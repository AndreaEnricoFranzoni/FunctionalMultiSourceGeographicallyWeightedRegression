199 inline
200 void
201 computeBNew(const std::map<std::string,std::vector< functional_matrix_diagonal<INPUT,OUTPUT> >> &W)
202 override
203 {
204 assert(W.size() == 1);
205 auto Wnc_new = W.at(std::string{fwr_FGWR_predictor<INPUT,OUTPUT>::id_NC});
206 for(std::size_t i = 0; i < Wnc_new.size(); ++i){
207 assert((Wnc_new[i].rows() == this->n_train()) && (Wnc_new[i].cols() == this->n_train()));}
208 //number of units to be predicted
209 std::size_t n_pred = Wnc_new.size();
210
211 //compute the non-stationary betas in the new locations
212 //penalties in the new locations
213 //(j_tilde + Rnc)^-1 (LncxLnc)
214 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_Rnc_inv = this->operator_comp().compute_penalty(m_eta_t,m_Xnc_train_t,Wnc_new,m_Xnc_train,m_eta,m_Rnc); //per applicarlo: j_double_tilde_RE_inv[i].solve(M) equivale a ([J_i_tilde_tilde + Re]^-1)*M
215 //COMPUTING all the m_bnc in the new locations, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE NON-STATIONARY BETAS (n_pred elements Lncx1)
216 m_bnc_pred = this->operator_comp().compute_operator(m_eta_t,m_Xnc_train_t,Wnc_new,m_y_train,j_Rnc_inv);
217
218 //
219 //wrapping the b from the shape useful for the computation into a more useful format for reporting the results
220 //
221 //non-stationary covariates (qnc elements of n_pred elements 1xLnc_i)
222 m_Bnc_pred = this->operator_comp().wrap_operator(m_bnc_pred,m_Lnc_j,m_qnc,n_pred);
223
224 }

239 inline
240 void
241 computeNonStationaryBetas()
242 override
243 {
244 std::size_t n_pred = m_bnc_pred.size();
245 m_BetaNC.resize(n_pred);
246
247#ifdef _OPENMP
248#pragma omp parallel for shared(m_BetaNC,m_eta,m_bnc_pred,n_pred) num_threads(this->number_threads())
249#endif
250 for(std::size_t i = 0; i < n_pred; ++i)
251 {
252 m_BetaNC[i] = fm_prod(m_eta,m_bnc_pred[i]);
253 }

262 inline
263 functional_matrix<INPUT,OUTPUT>
264 predict(const std::map<std::string,functional_matrix<INPUT,OUTPUT>> &X_new)
265 const
266 override
267 {
268 //input coherence
269 assert(X_new.size() == 1);
270
271 auto Xnc_new = X_new.at(std::string{fwr_FGWR_predictor<INPUT,OUTPUT>::id_NC});
272
273 std::size_t n_pred = Xnc_new.rows();
274 assert(n_pred == m_BetaNC.size());
275 assert(Xnc_new.cols() == m_qnc);
276
277
278 //y_new = X_new*beta = Xnc_new*beta_nc
279 functional_matrix<INPUT,OUTPUT> y_new_NC(n_pred,1);
280
281
282#ifdef _OPENMP
283#pragma omp parallel for shared(Xnc_new,m_BetaNC,y_new_NC,n_pred,m_qnc) num_threads(this->number_threads())
284#endif
285 for(std::size_t i = 0; i < n_pred; ++i)
286 {
287 std::vector< FUNC_OBJ<INPUT,OUTPUT> > xnc_new_i(Xnc_new.row(i).cbegin(),Xnc_new.row(i).cend()); //1xqnc
288 functional_matrix<INPUT,OUTPUT> Xnc_new_i(xnc_new_i,1,m_qnc);
289 y_new_NC(i,0) = fm_prod(Xnc_new_i,m_BetaNC[i],this->number_threads())(0,0);
290 }
291
292 return y_new_NC;
293 }
