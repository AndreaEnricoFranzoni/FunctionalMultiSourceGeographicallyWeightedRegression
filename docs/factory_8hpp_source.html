177 template <typename AbstractProduct, typename Identifier, typename Builder>
178 template <typename... Args>
179 std::unique_ptr<AbstractProduct>
180 Factory<AbstractProduct,Identifier,Builder>::create(Identifier const & name,
181 Args &&...args)
182 const
183 {
184
185 auto f = _storage.find(name); //C++11
186 //object not registered: exception
187 if (f == _storage.end())
188 {
189 std::string out="Identifier " + name + " is not stored in the factory";
190 throw std::invalid_argument(out);
191 }
192 else
193 {
194 return f->second(std::forward<Args>(args)...);
195 }
196 }

210 Factory<AbstractProduct,Identifier,Builder>::add(Identifier const & name, Builder_type const & func)
211 {
212 auto f = _storage.insert(std::make_pair(name, func));
213 if (f.second == false)
214 throw std::invalid_argument("Double registration in Factory");

228 Factory<AbstractProduct,Identifier,Builder>::registered()
229 const
230 {
231 std::vector<Identifier> tmp;
232 tmp.reserve(_storage.size());
233 for(auto i=_storage.begin(); i!=_storage.end();++i)
234 tmp.push_back(i->first);
235 return tmp;
