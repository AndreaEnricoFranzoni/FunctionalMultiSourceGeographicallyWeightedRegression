45template <typename Triangulation_, typename CoordsMatrix_>
46 //requires(internals::is_eigen_dense_xpr_v<CoordsMatrix_>)
47 requires(fdagwr_concepts::as_interval<Triangulation_> && fdapde::internals::is_eigen_dense_xpr_v<CoordsMatrix_>)
48inline
49Eigen::SparseMatrix<double>
50bsplines_basis_evaluation(const fdapde::BsSpace<Triangulation_>& bs_space,
51 CoordsMatrix_&& coords)
52{
53 static constexpr int embed_dim = Triangulation_::embed_dim;
54 assert(coords.rows() > 0 && coords.cols() == embed_dim);
55
56 int n_shape_functions = bs_space.n_shape_functions();
57 int n_dofs = bs_space.n_dofs();
58 int n_locs = coords.rows();
59 Eigen::SparseMatrix<double> psi_(n_locs, n_dofs);
60 std::vector<Eigen::Triplet<double>> triplet_list;
61 triplet_list.reserve(n_locs * n_shape_functions);
62
63 Eigen::Matrix<int, Eigen::Dynamic, 1> cell_id = bs_space.triangulation().locate(coords);
64 const auto& dof_handler = bs_space.dof_handler();
65 // build basis evaluation matrix
66 for (int i = 0; i < n_locs; ++i) {
67 if (cell_id[i] != -1) { // point falls inside domain
68 Eigen::Matrix<double, embed_dim, 1> p_i(coords.row(i));
69 auto cell = dof_handler.cell(cell_id[i]);
70 // update matrix
71 for (std::size_t h = 0; h < cell.dofs().size(); ++h) {
72 int active_dof = cell.dofs()[h];
73 triplet_list.emplace_back(i, active_dof, bs_space.eval_cell_value(active_dof, p_i)); // \psi_j(p_i)
74 }
75 }
76 }
77 // finalize construction
78 psi_.setFromTriplets(triplet_list.begin(), triplet_list.end());
79 psi_.makeCompressed();
80 return psi_;
81}

211 override

214
215 //psi: an knots.size() x number of basis matrix: each row represents a knot, every column a basis: contains its evaluation
216 Eigen::SparseMatrix<double> psi = this->eval_base_on_locs(knots);
217 Eigen::SparseQR<Eigen::SparseMatrix<double>, Eigen::COLAMDOrdering<int>> solver;
218 //performs (t(Psi)*Psi)^(-1) * t(Psi)
219 solver.compute(psi);
220
221 return solver.solve(f_ev);
