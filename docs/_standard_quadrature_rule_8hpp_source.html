177template <unsigned int N>
178double
179apsc::NumericalIntegration::StandardQuadratureRule<N>::apply(
180 FunPoint const &f, double const &a, double const &b) const
181{
182 double h2=(b - a) * 0.5; // half length
183 double xm=(a + b) * 0.5; // midpoint
184 // scale functions
185 auto fscaled = [&h2,&xm,&f](double x) { return f(x * h2 + xm); };
186 double tmp=0.0;
187 auto np = n_.begin();
188 // for (auto wp=w_.begin();wp<w_.end();++wp,++np)
189 for(auto weight : w_)
190 {
191 tmp += fscaled(*(np)) * weight;
192 ++np;
193 }
194
195 return h2 * tmp;
196}
