127 double apply(FunPoint const &f) const
128 {
129 double result(0);
130#ifndef PARALLELCPP
131#ifdef _OPENMP
132#pragma omp parallel for reduction(+ : result) shared(f)
133#endif
134 for(unsigned int i = 0u; i < mesh_.numNodes() - 1; ++i)
135 {
136 double const a = mesh_[i];
137 double const b = mesh_[i + 1];
138 result += rule_->apply(f, a, b);
139 }
140#else
141 result = std::transform_reduce(std::execution::par_unseq, mesh_.begin(),
142 mesh_.end() - 1, 0.0, std::plus<double>(),
143 [this, &f](double const a, double const b) {
144 return this->rule_->apply(f, a, b);
145 });
146#endif
147 return result;
148 }
