305 inline
306 void
307 compute()
308 override
309 {
310 //exact estimation
311 if(!this->in_cascade_estimation())
312 {
313 //(j_tilde_tilde + Re)^-1 (LexLe)
314 std::cout << "Computing (j_tilde_tilde + Re)^-1" << std::endl;
315 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_double_tilde_Re_inv = this->operator_comp().compute_penalty(m_theta_t,m_Xe_t,m_We,m_Xe,m_theta,m_Re); //to apply: j_double_tilde_RE_inv[i].solve(M) is equal to ([J_i_tilde_tilde + Re]^-1)*M
316 //A_E_i (n elements Lex(n*Ly))
317 std::cout << "Computing A_e_i" << std::endl;
318 m_A_e = this->operator_comp().compute_operator(m_theta_t,m_Xe_t,m_We,m_phi,j_double_tilde_Re_inv);
319 //H_e(t) (nx(n*Ly))
320 std::cout << "Computing H_e(t)" << std::endl;
321 functional_matrix<INPUT,OUTPUT> H_e = this->operator_comp().compute_functional_operator(m_Xe,m_theta,m_A_e);
322 //B_E_i (n elements LexLc)
323 std::cout << "Computing B_e_i" << std::endl;
324 m_B_e = this->operator_comp().compute_operator(m_theta_t,m_Xe_t,m_We,m_Xc,m_omega,j_double_tilde_Re_inv);
325 //K_e_c(t) (nxLc)
326 std::cout << "Computing K_e_c(t)" << std::endl;
327 functional_matrix<INPUT,OUTPUT> K_e_c = this->operator_comp().compute_functional_operator(m_Xe,m_theta,m_B_e);
328 //B_E_i_for_K_e_s (n elements LexLs)
329 std::cout << "Computing B_e_i_for_K_e_S" << std::endl;
330 m_B_e_for_K_e_s = this->operator_comp().compute_operator(m_theta_t,m_Xe_t,m_We,m_Xs,m_psi,j_double_tilde_Re_inv);
331 //K_e_s(t) (nxLs)
332 std::cout << "Computing K_e_s(t)" << std::endl;
333 functional_matrix<INPUT,OUTPUT> K_e_s = this->operator_comp().compute_functional_operator(m_Xe,m_theta,m_B_e_for_K_e_s);
334 //X_s_crossed(t) (nxLs)
335 functional_matrix<INPUT,OUTPUT> X_s_crossed = fm_prod(m_Xs,m_psi) - K_e_s;
336 functional_matrix<INPUT,OUTPUT> X_s_crossed_t = X_s_crossed.transpose();
337 //(j_tilde + Rs)^-1 (LsxLs)
338 std::cout << "Computing (j_tilde + Rs)^-1" << std::endl;
339 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_tilde_Rs_inv = this->operator_comp().compute_penalty(X_s_crossed_t,m_Ws,X_s_crossed,m_Rs);
340
341 //A_S_i (n elements Lsx(n*Ly))
342 functional_matrix<INPUT,OUTPUT> rhs_As = m_phi - H_e;
343 std::cout << "Computing A_s_i" << std::endl;
344 m_A_s = this->operator_comp().compute_operator(X_s_crossed_t,m_Ws,rhs_As,j_tilde_Rs_inv);
345 //H_s(t) (nx(n*Ly))
346 std::cout << "Computing H_s(t)" << std::endl;
347 functional_matrix<INPUT,OUTPUT> H_s = this->operator_comp().compute_functional_operator(m_Xs,m_psi,m_A_s);
348 //A_ES_i (n elements Lex(n*Ly))
349 std::cout << "Computing A_es_i" << std::endl;
350 m_A_es = this->operator_comp().compute_operator(m_theta_t,m_Xe_t,m_We,H_s,j_double_tilde_Re_inv);
351 //H_es(t) (nx(n*Ly))
352 std::cout << "Computing H_es(t)" << std::endl;
353 functional_matrix<INPUT,OUTPUT> H_es = this->operator_comp().compute_functional_operator(m_Xe,m_theta,m_A_es);
354
355 //B_S_i (n elements LsxLc)
356 functional_matrix<INPUT,OUTPUT> rhs_Bs = fm_prod(m_Xc,m_omega) - K_e_c;
357 std::cout << "Computing B_s_i" << std::endl;
358 m_B_s = this->operator_comp().compute_operator(X_s_crossed_t,m_Ws,rhs_Bs,j_tilde_Rs_inv);
359 //K_s_c(t) (nXLc)
360 std::cout << "Computing K_s_c(t)" << std::endl;
361 functional_matrix<INPUT,OUTPUT> K_s_c = this->operator_comp().compute_functional_operator(m_Xs,m_psi,m_B_s);
362 //B_ES_i (n elements LexLc)
363 std::cout << "Computing B_es_i" << std::endl;
364 m_B_es = this->operator_comp().compute_operator(m_theta_t,m_Xe_t,m_We,K_s_c,j_double_tilde_Re_inv);
365 //K_es_c(t) (nxLc)
366 std::cout << "Computing K_es_c(t)" << std::endl;
367 functional_matrix<INPUT,OUTPUT> K_es_c = this->operator_comp().compute_functional_operator(m_Xe,m_theta,m_B_es);
368
369 //y_new(t) (nx1)
370 std::cout << "Computing y_new(t)" << std::endl;
371 functional_matrix<INPUT,OUTPUT> y_new = fm_prod(functional_matrix<INPUT,OUTPUT>(m_phi - H_e - H_s + H_es),m_c,this->number_threads());
372 //X_c_crossed (nxqc)
373 functional_matrix<INPUT,OUTPUT> X_c_crossed = fm_prod(m_Xc,m_omega) - K_e_c - K_s_c + K_es_c;
374 functional_matrix<INPUT,OUTPUT> X_c_crossed_t = X_c_crossed.transpose();
375 //[J + Rc]^-1 (LcxLc)
376 std::cout << "Computing (j + Rc)^-1" << std::endl;
377 Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> j_Rc_inv = this->operator_comp().compute_penalty(X_c_crossed_t,m_Wc,X_c_crossed,m_Rc);
378
379
380 //COMPUTING m_bc, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATIONARY BETAS (Lcx1)
381 std::cout << "Computing bc" << std::endl;
382 m_bc = this->operator_comp().compute_operator(X_c_crossed_t,m_Wc,y_new,j_Rc_inv);
383
384
385 //y_tilde_hat(t) (nx1)
386 std::cout << "Computing y_tilde_hat(t)" << std::endl;
387 functional_matrix<INPUT,OUTPUT> y_tilde_hat = m_y - fm_prod(fm_prod(m_Xc,m_omega),m_bc,this->number_threads());
388 //c_tilde_hat: smoothing on y_tilde_hat(t) with respect of the basis of y ((n*Ly)x1)
389 std::cout << "Computing c_tilde_hat" << std::endl;
390 m_c_tilde_hat = columnize_coeff_resp(fm_smoothing<INPUT,OUTPUT,FDAGWR_TRAITS::basis_geometry>(y_tilde_hat,*m_basis_y,m_knots_smoothing));
391 //y_tilde_new(t) (nx1)
392 std::cout << "Computing y_tilde_new(t)" << std::endl;
393 functional_matrix<INPUT,OUTPUT> y_tilde_new = fm_prod(functional_matrix<INPUT,OUTPUT>(m_phi - H_e),m_c_tilde_hat,this->number_threads());
394
395
396 //COMPUTING all the m_bs, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATION-DEPENDENT BETAS (n elements Lsx1)
397 std::cout << "Computing bs" << std::endl;
398 m_bs = this->operator_comp().compute_operator(X_s_crossed_t,m_Ws,y_tilde_new,j_tilde_Rs_inv);
399
400
401 //y_tilde_tilde_hat(t) (nx1)
402 std::cout << "Computing y_tilde_tilde_hat(t)" << std::endl;
403 functional_matrix<INPUT,OUTPUT> y_tilde_tilde_hat = y_tilde_hat - this->operator_comp().compute_functional_operator(m_Xs,m_psi,m_bs);
404
405
406 //COMPUTING all the m_be, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE EVENT-DEPENDENT BETAS (n elementd Lex1)
407 std::cout << "Computing be" << std::endl;
408 m_be = this->operator_comp().compute_operator(m_theta_t,m_Xe_t,m_We,y_tilde_tilde_hat,j_double_tilde_Re_inv);
409 }
410 //in cascade estimation
411 else
412 {
413 //[J + Rc]^-1 (LcxLc)
414 std::cout << "Computing (j + Rc)^-1" << std::endl;
415 Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> j_Rc_inv = this->operator_comp().compute_penalty(m_omega_t,m_Xc_t,m_Wc,m_Xc,m_omega,m_Rc);
416
417 //COMPUTING m_bc, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATIONARY BETAS (Lcx1)
418 std::cout << "Computing bc" << std::endl;
419 m_bc = this->operator_comp().compute_operator(m_omega_t,m_Xc_t,m_Wc,m_y,j_Rc_inv);
420
421 //y_tilde_hat (nx1)
422 functional_matrix<INPUT,OUTPUT> y_tilde_hat = m_y - fm_prod(fm_prod(m_Xc,m_omega),m_bc,this->number_threads());
423
424 //[J_i + Rs]^-1 (LsxLs)
425 std::cout << "Computing (j + Rs)^-1" << std::endl;
426 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_i_Rs_inv = this->operator_comp().compute_penalty(m_psi_t,m_Xs_t,m_Ws,m_Xs,m_psi,m_Rs);
427 //COMPUTING m_bs, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATIONARY BETAS (n elements Lsx1)
428 std::cout << "Computing bs" << std::endl;
429 m_bs = this->operator_comp().compute_operator(m_psi_t,m_Xs_t,m_Ws,y_tilde_hat,j_i_Rs_inv);
430
431 //y_tilde_tilde_hat (nx1)
432 functional_matrix<INPUT,OUTPUT> y_tilde_tilde_hat(this->n(),1);
433
434 //extra objects (default values of 0s)
435 m_A_e.resize(this->n());
436 m_B_e_for_K_e_s.resize(this->n());
437
438#ifdef _OPENMP
439#pragma omp parallel for shared(y_tilde_tilde_hat,m_Xs,m_psi,y_tilde_hat) num_threads(this->number_threads())
440#endif
441 for(std::size_t i = 0; i < this->n(); ++i)
442 {
443 std::vector< FUNC_OBJ<INPUT,OUTPUT> > xs_i(m_Xs.row(i).cbegin(),m_Xs.row(i).cend()); //1xqs
444 functional_matrix<INPUT,OUTPUT> Xs_i(xs_i,1,m_qs);
445 functional_matrix<INPUT,OUTPUT> y_tilde_hat_i(1,1,y_tilde_hat(i,0));
446 y_tilde_tilde_hat(i,0) = (y_tilde_hat_i - fm_prod(fm_prod(Xs_i,m_psi),m_bs[i],this->number_threads()))(0,0);
447
448 //default values of 0 for returning elements
449 m_A_e[i] = FDAGWR_TRAITS::Dense_Matrix::Zero(m_Le,m_Ly*this->n());
450 m_B_e_for_K_e_s[i] = FDAGWR_TRAITS::Dense_Matrix::Zero(m_Le,m_Ls);
451 }
452
453 //[J_i + Re]^-1 (LexLe)
454 std::cout << "Computing (j + Re)^-1" << std::endl;
455 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_i_Re_inv = this->operator_comp().compute_penalty(m_theta_t,m_Xe_t,m_We,m_Xe,m_theta,m_Re);
456 //COMPUTING m_be, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATIONARY BETAS (n elements Lex1)
457 std::cout << "Computing be" << std::endl;
458 m_be = this->operator_comp().compute_operator(m_theta_t,m_Xe_t,m_We,y_tilde_tilde_hat,j_i_Re_inv);
459
460 //default values of 0 for returning elements
461 m_c_tilde_hat = FDAGWR_TRAITS::Dense_Matrix::Zero(m_Ly*this->n(),1);
462 }
463
464
465 //
466 //wrapping the b from the shape useful for the computation into a more useful format
467 //
468 //stationary covariates (qc elements 1xLc_i)
469 m_Bc = this->operator_comp().wrap_operator(m_bc,m_Lc_j,m_qc);
470 //event-dependent covariates (qe elements of n elements 1xLe_i)
471 m_Be = this->operator_comp().wrap_operator(m_be,m_Le_j,m_qe,this->n());
472 //station-dependent covariates (qs elements of n elements 1xLs_i)
473 m_Bs = this->operator_comp().wrap_operator(m_be,m_Ls_j,m_qs,this->n());
474 }
