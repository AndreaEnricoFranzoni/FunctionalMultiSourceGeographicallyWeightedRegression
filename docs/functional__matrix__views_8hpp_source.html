44{
46 using value_type = typename std::remove_pointer<Ptr>::type;
48 using difference_type = std::ptrdiff_t;
50 using pointer = Ptr;
52 using reference = typename std::add_lvalue_reference<value_type>::type;
54 using iterator_category = std::forward_iterator_tag;

59 std::ptrdiff_t step;

66 StridedIterator(Ptr p, std::ptrdiff_t s) : ptr(p), step(s) {}

72 reference operator*() const { return *ptr; }

78 StridedIterator& operator++() { ptr += step; return *this; }
