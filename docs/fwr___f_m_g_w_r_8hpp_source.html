240 inline
241 void
242 compute()
243 override
244 {
245 //exact estimation
246 if(!this->in_cascade_estimation())
247 {
248 //(j_tilde + Rnc)^-1 (LncxLnc)
249 std::cout << "Computing (j_tilde + Rnc)^-1" << std::endl;
250 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_tilde_Rnc_inv = this->operator_comp().compute_penalty(m_eta_t,m_Xnc_t,m_Wnc,m_Xnc,m_eta,m_Rnc); //per applicarlo: j_double_tilde_RE_inv[i].solve(M) equivale a ([J_i_tilde_tilde + Re]^-1)*M
251 //A_NC_i (n elements Lncx(n*Ly))
252 std::cout << "Computing A_nc_i" << std::endl;
253 m_A_nc = this->operator_comp().compute_operator(m_eta_t,m_Xnc_t,m_Wnc,m_phi,j_tilde_Rnc_inv);
254 //H_nc(t) (nx(n*Ly))
255 std::cout << "Computing H_nc(t)" << std::endl;
256 functional_matrix<INPUT,OUTPUT> H_nc = this->operator_comp().compute_functional_operator(m_Xnc,m_eta,m_A_nc);
257 //B_NC_i (n elements LncxLc)
258 std::cout << "Computing B_nc_i" << std::endl;
259 m_B_nc = this->operator_comp().compute_operator(m_eta_t,m_Xnc_t,m_Wnc,m_Xc,m_omega,j_tilde_Rnc_inv);
260 //K_nc_c(t) (nxLc)
261 std::cout << "Computing K_nc_c(t)" << std::endl;
262 functional_matrix<INPUT,OUTPUT> K_nc_c = this->operator_comp().compute_functional_operator(m_Xnc,m_eta,m_B_nc);
263
264 //y_new(t) (nx1)
265 std::cout << "Computing y_new(t)" << std::endl;
266 functional_matrix<INPUT,OUTPUT> y_new = fm_prod(functional_matrix<INPUT,OUTPUT>(m_phi - H_nc),m_c,this->number_threads());
267 //Xc_crossed (nxqnc)
268 functional_matrix<INPUT,OUTPUT> X_c_crossed = fm_prod(m_Xc,m_omega) - K_nc_c;
269 functional_matrix<INPUT,OUTPUT> X_c_crossed_t = X_c_crossed.transpose();
270 //[J + Rc]^-1 (LcxLc)
271 std::cout << "Computing (j + Rc)^-1" << std::endl;
272 Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> j_Rc_inv = this->operator_comp().compute_penalty(X_c_crossed_t,m_Wc,X_c_crossed,m_Rc);
273
274
275 //COMPUTING m_bc, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATIONARY BETAS (Lcx1)
276 std::cout << "Computing bc" << std::endl;
277 m_bc = this->operator_comp().compute_operator(X_c_crossed_t,m_Wc,y_new,j_Rc_inv);
278
279 //y_tilde_hat(t) (nx1)
280 std::cout << "Computing y_tilde_hat(t)" << std::endl;
281 functional_matrix<INPUT,OUTPUT> y_tilde_hat = m_y - fm_prod(fm_prod(m_Xc,m_omega),m_bc,this->number_threads());
282 //c_tilde_hat: smoothing on y_tilde_hat(t) with respect of the basis of y ((n*Ly)x1)
283 std::cout << "Computing c_tilde_hat" << std::endl;
284 m_c_tilde_hat = columnize_coeff_resp(fm_smoothing<INPUT,OUTPUT,FDAGWR_TRAITS::basis_geometry>(y_tilde_hat,*m_basis_y,m_knots_smoothing));
285 //y_tilde_new(t) (nx1)
286 std::cout << "Computing y_tilde_new(t)" << std::endl;
287 functional_matrix<INPUT,OUTPUT> y_tilde_new = fm_prod(m_phi,m_c_tilde_hat);
288
289
290 //COMPUTING all the m_bnc, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATION-DEPENDENT BETAS (n elements Lncx1)
291 std::cout << "Computing bnc" << std::endl;
292 m_bnc = this->operator_comp().compute_operator(m_eta_t,m_Xnc_t,m_Wnc,y_tilde_new,j_tilde_Rnc_inv);
293 }
294 //in cascade estimation
295 else
296 {
297 //[J + Rc]^-1 (LcxLc)
298 std::cout << "Computing (j + Rc)^-1" << std::endl;
299 Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> j_Rc_inv = this->operator_comp().compute_penalty(m_omega_t,m_Xc_t,m_Wc,m_Xc,m_omega,m_Rc);
300 //COMPUTING m_bc, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATIONARY BETAS (Lcx1)
301 std::cout << "Computing bc" << std::endl;
302 m_bc = this->operator_comp().compute_operator(m_omega_t,m_Xc_t,m_Wc,m_y,j_Rc_inv);
303 //y_tilde (nx1)
304 functional_matrix<INPUT,OUTPUT> y_tilde = m_y - fm_prod(fm_prod(m_Xc,m_omega),m_bc,this->number_threads());
305 //[J_i + Rnc]^-1 (LncxLnc)
306 std::cout << "Computing (j + Rnc)^-1" << std::endl;
307 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_i_Rnc_inv = this->operator_comp().compute_penalty(m_eta_t,m_Xnc_t,m_Wnc,m_Xnc,m_eta,m_Rnc);
308 //COMPUTING m_bnc, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATIONARY BETAS (n elements Lncx1)
309 std::cout << "Computing bnc" << std::endl;
310 m_bnc = this->operator_comp().compute_operator(m_eta_t,m_Xnc_t,m_Wnc,y_tilde,j_i_Rnc_inv);
311 //extra (default values of 0s)
312 m_c_tilde_hat = Eigen::MatrixXd::Zero(m_Ly*this->n(),1);
313 }
314
315
316 //
317 //wrapping the b from the shape useful for the computation into a more useful format
318 //
319 //stationary covariates (qc elements 1xLc_i)
320 m_Bc = this->operator_comp().wrap_operator(m_bc,m_Lc_j,m_qc);
321 //non-stationary covariates (qnc elements of n elements 1xLnc_i)
322 m_Bnc = this->operator_comp().wrap_operator(m_bnc,m_Lnc_j,m_qnc,this->n());
323 }
