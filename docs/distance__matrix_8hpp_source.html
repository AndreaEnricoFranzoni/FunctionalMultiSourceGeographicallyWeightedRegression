107 template<typename COORDINATES_OBJ>
108 distance_matrix(COORDINATES_OBJ&& coordinates,
109 int number_threads)
110 :
111 m_number_locations(coordinates.rows()), //pass the number of statistical units
112 m_coordinates{std::forward<COORDINATES_OBJ>(coordinates)}, //pass the coordinates
113 m_flag_comp_dist(m_number_locations > 0), //if there are locations
114 m_num_threads(number_threads) //number of threads for paralelization
115 {
116 //cheack the correct dimension of the coordinates matrix
117 assert((void("Coordinates matrix has to have 2 columns"), coordinates.cols() == 2));
118 //the number of distances to be computed is m*(m+1)/2
119 if (m_flag_comp_dist) m_number_dist_comp = (m_number_locations*(m_number_locations + static_cast<std::size_t>(1)))/static_cast<std::size_t>(2);
120 }

149 inline
150 double
151 operator()(std::size_t i, std::size_t j)
152 const
153 {
154 if (i < j) std::swap(i, j);
155 return m_distances[i*(i+1)/2 + j];
156 }

164 inline
165 FDAGWR_TRAITS::Dense_Vector
166 operator[](std::size_t col_i)
167 const
168 {
169 //cheack the correct dimension of the coordinates matrix
170 assert((void("The column index has to be in {0,1,...,number-of-statistical-units - 1}"),
171 0<=col_i && col_i < m_number_locations));
172
173 //container for the column
174 FDAGWR_TRAITS::Dense_Vector column(m_number_locations);
175
176 //vector to store the indeces of the elements of m_distances that refers to a distance computed with respect to the i-th statistical unit
177 std::vector<std::size_t> access_indeces;
178 //there is a total of number-of-statistical-units distances computed for each statistical unit
179 access_indeces.reserve(m_number_locations);
180
181 //the first index is the index of start of the column (elem of the column in the first row)
182 access_indeces.emplace_back(distance_matrix::partial_sum(col_i));
183
184 //then, the following indeces are the colomn_number integers following the first index (elems of the column up until diagonal)
185 for (size_t i = 1; i <= col_i; ++i){ access_indeces.emplace_back(access_indeces.front() + i);}
186
187 //last set of indeces: elements of row col_i from the diagonal up until the end of the row
188 //(index of the previous element plus the column index of the next one in the same row)
189 for (std::size_t i = col_i + 1; i < m_number_locations; ++i){ access_indeces.emplace_back(access_indeces.back()+i);}
190
191 //filling the column
192#ifdef _OPENMP
193#pragma omp parallel for shared(m_number_locations,access_indeces) num_threads(m_num_threads)
194#endif
195 for(std::size_t i = 0; i < m_number_locations; ++i){ column(i) = m_distances[access_indeces[i]];}
196
197 return column;
198 }
