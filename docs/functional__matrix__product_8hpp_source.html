59template< typename INPUT = double, typename OUTPUT = double >
60 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>)
61inline
62functional_matrix<INPUT,OUTPUT>
63fm_prod(const functional_matrix<INPUT,OUTPUT> &M1,
64 const functional_matrix<INPUT,OUTPUT> &M2,
65 int number_threads)
66{
67 //checking matrices dimensions
68 if (M1.cols() != M2.rows())
69 throw std::invalid_argument("Incompatible matrix dimensions for functional matrix product");
70
71 //std::function object stored
72 using F_OBJ = FUNC_OBJ<INPUT,OUTPUT>;
73 //std::function input type
74 using F_OBJ_INPUT = fm_utils::input_param_t<F_OBJ>;
75 //initial point for f_sum
76 F_OBJ f_null = [](F_OBJ_INPUT x){return static_cast<OUTPUT>(0);};
77 //reducing operation for transform_reduce
78 std::function<F_OBJ(F_OBJ,F_OBJ)> f_sum = [](F_OBJ f1, F_OBJ f2){return [f1,f2](F_OBJ_INPUT x){return f1(x)+f2(x);};};
79 //binary operation for transform_reduce
80 std::function<F_OBJ(F_OBJ,F_OBJ)> f_prod = [](F_OBJ f1, F_OBJ f2){return [f1,f2](F_OBJ_INPUT x){return f1(x)*f2(x);};};
81
82 //resulting matrix
83 functional_matrix<INPUT,OUTPUT> prod(M1.rows(),M2.cols());
84
85#ifdef _OPENMP
86#pragma omp parallel for collapse(2) shared(M1,M2,prod,f_null,f_sum,f_prod) num_threads(number_threads)
87#endif
88 for(std::size_t i = 0; i < prod.rows(); ++i){
89 for(std::size_t j = 0; j < prod.cols(); ++j){
90 //dot product within the row i-th of M1 and the col j-th of M2: using the views, access to row and cols is O(1)
91 prod(i,j) = std::transform_reduce(M1.row(i).cbegin(),
92 M1.row(i).cend(),
93 M2.col(j).cbegin(),
94 f_null, //initial value (null function)
95 f_sum, //reduce operation
96 f_prod);}} //transform operation within the two ranges
97
98 return prod;
99}

112template< typename INPUT = double, typename OUTPUT = double >
113 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>)
114inline
115functional_matrix<INPUT,OUTPUT>
116fm_prod(const functional_matrix<INPUT,OUTPUT> &M1,
117 const functional_matrix_sparse<INPUT,OUTPUT> &SM2)
118{
119 //checking matrices dimensions
120 if (M1.cols() != SM2.rows())
121 throw std::invalid_argument("Incompatible matrix dimensions for functional matrix product");
122
123 //std::function object stored
124 using F_OBJ = FUNC_OBJ<INPUT,OUTPUT>;
125 //std::function input type
126 using F_OBJ_INPUT = fm_utils::input_param_t<F_OBJ>;
127
128 //initial point for f_sum
129 F_OBJ f_null = [](F_OBJ_INPUT x){return static_cast<OUTPUT>(0);};
130 //reducing operation for transform_reduce
131 std::function<F_OBJ(F_OBJ,F_OBJ)> f_sum = [](F_OBJ f1, F_OBJ f2){return [f1,f2](F_OBJ_INPUT x){return f1(x)+f2(x);};};
132 //binary operation for transform_reduce
133 std::function<F_OBJ(F_OBJ,F_OBJ)> f_prod = [](F_OBJ f1, F_OBJ f2){return [f1,f2](F_OBJ_INPUT x){return f1(x)*f2(x);};};
134
135 //resulting matrix
136 functional_matrix<INPUT,OUTPUT> prod(M1.rows(),SM2.cols(),f_null);
137
138
139 //fixing the col in the sparse matrix, computing all the products with that columns
140 for(std::size_t j = 0; j < prod.cols(); ++j){
141 //if there is at least one element in col j-th
142 if(SM2.check_col_presence(j)){
143 //computing all the products with col j-th
144 for(std::size_t i = 0; i < prod.rows(); ++i){
145 //products and summations are computed only with respect to the non-null elements
146 for(std::size_t ii = SM2.cols_idx()[j]; ii < SM2.cols_idx()[j+1]; ++ii){
147 //row that contains an element in col j-th
148 std::size_t row_col_j = SM2.rows_idx()[ii];
149 prod(i,j) = f_sum(prod(i,j),f_prod(M1(i,row_col_j),SM2(row_col_j,j)));}}}}
150
151 return prod;
152}

165template< typename INPUT = double, typename OUTPUT = double >

168functional_matrix<INPUT,OUTPUT>
169fm_prod(const functional_matrix_sparse<INPUT,OUTPUT> &SM1,

187
188 //resulting matrix
189 functional_matrix<INPUT,OUTPUT> prod(SM1.rows(),M2.cols(),f_null);
190
191 //loop su tutte le colonne di SM1 perch√® la matrice sparsa va passata columnwise
192 for(std::size_t j_s = 0; j_s < SM1.cols(); ++j_s){
193 //loop sulle righe non-nulle della colonna j-th
194 for(std::size_t row_i_idx = SM1.cols_idx()[j_s]; row_i_idx < SM1.cols_idx()[j_s+1]; ++row_i_idx){
195 //row not null
196 std::size_t i = SM1.rows_idx()[row_i_idx];
197 //updating step-by-step that element in the product
198 for (std::size_t j = 0; j < prod.cols(); ++j){
199 prod(i,j) = f_sum( prod(i,j), f_prod(SM1(i,j_s),M2(j_s,j)) );}}}

201 return prod;
