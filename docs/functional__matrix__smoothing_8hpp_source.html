51template< typename INPUT = double, typename OUTPUT = double, class domain_type = FDAGWR_TRAITS::basis_geometry >
52 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>) && fdagwr_concepts::as_interval<domain_type>
53inline
54FDAGWR_TRAITS::Dense_Matrix
55f_smoothing(const FUNC_OBJ<INPUT,OUTPUT> &f,
56 const basis_base_class<domain_type> & basis,
57 const FDAGWR_TRAITS::Dense_Matrix & knots)
58{
59 //evaluating the function all along the knots
60 FDAGWR_TRAITS::Dense_Matrix f_ev(knots.rows(),1);
61 for(std::size_t i = 0; i < knots.rows(); ++i){ f_ev(i,0) = f(knots(i,0));}
62
63 return basis.smoothing(f_ev,knots);
64}

76template< typename INPUT = double, typename OUTPUT = double, class domain_type = FDAGWR_TRAITS::basis_geometry >
77 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>) && fdagwr_concepts::as_interval<domain_type>
78inline
79FDAGWR_TRAITS::Dense_Matrix
80fm_smoothing(const functional_matrix<INPUT,OUTPUT> &fm,
81 const basis_base_class<domain_type> & basis,
82 const FDAGWR_TRAITS::Dense_Matrix & knots)
83{
84 assert(fm.cols() == 1);
85
86 std::size_t num_basis = basis.number_of_basis();
87 std::size_t num_f = fm.rows();
88
89 //the result is num basis x num functions
90 FDAGWR_TRAITS::Dense_Matrix c(num_basis,num_f);
91
92 for (std::size_t i = 0; i < num_f; ++i){ c.col(i) = f_smoothing(fm(i,0),basis,knots).col(0);}
93
94 return c; //num basis x num functions
95}
