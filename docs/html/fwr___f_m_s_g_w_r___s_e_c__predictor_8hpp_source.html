283 inline
284 void
285 computePartialResiduals()
286 override
287 {
288 //exact estimation
289 if (!this->in_cascade_estimation())
290 {
291 //retrieve the partial residuals from the fitted model
292 //K_s_e(t) (n_train x Le)
293 m_K_s_e = this->operator_comp().compute_functional_operator(m_Xs_train,m_psi,m_B_s_for_K_s_e);
294 //X_e_crossed(t) (n_train x Le)
295 m_Xe_train_crossed = fm_prod(m_Xe_train,m_theta) - m_K_s_e;
296 m_Xe_train_crossed_t = m_Xe_train_crossed.transpose();
297 //y_tilde_hat(t) (n_trainx1)
298 m_y_tilde_hat = fm_prod(m_phi,m_c_tilde_hat);
299 //Hs(t) (n_trainx(n_train*Ly))
300 m_H_s = this->operator_comp().compute_functional_operator(m_Xs_train,m_psi,m_A_s);
301 //y_tilde_new(t) (n_trainx1)
302 m_y_tilde_new = fm_prod(functional_matrix<INPUT,OUTPUT>(m_phi - m_H_s),m_c_tilde_hat,this->number_threads());
303 //y_tilde_tilde_hat(t) (n_trainx1)
304 m_y_tilde_tilde_hat = m_y_tilde_hat - this->operator_comp().compute_functional_operator(m_Xe_train,m_theta,m_be_fitted);
305 }
306 //in cascade estimation
307 else
308 {
309 //y_tilde_hat(t) (n_trainx1)
310 m_y_tilde_hat = m_y_train - fm_prod(fm_prod(m_Xc_train,m_omega),m_bc_fitted,this->number_threads());
311 //y_tilde_tilde_hat(t) (n_trainx1)
312 m_y_tilde_tilde_hat = m_y_train;
313
314#ifdef _OPENMP
315#pragma omp parallel for shared(m_y_tilde_tilde_hat,m_Xe_train,m_theta,m_y_tilde_hat) num_threads(this->number_threads())
316#endif
317 for(std::size_t i = 0; i < this->n_train(); ++i)
318 {
319 std::vector< FUNC_OBJ<INPUT,OUTPUT> > xe_i(m_Xe_train.row(i).cbegin(),m_Xe_train.row(i).cend()); //1xqe
320 functional_matrix<INPUT,OUTPUT> Xe_i(xe_i,1,m_qe);
321 functional_matrix<INPUT,OUTPUT> y_tilde_hat_i(1,1,m_y_tilde_hat(i,0));
322 m_y_tilde_tilde_hat(i,0) = (y_tilde_hat_i - fm_prod(fm_prod(Xe_i,m_theta),m_be_fitted[i],this->number_threads()))(0,0);
323 }
324 }

332 inline
333 void
334 computeBNew(const std::map<std::string,std::vector< functional_matrix_diagonal<INPUT,OUTPUT> >> &W)
335 override
336 {
337 assert(W.size() == 2);
338 auto We_new = W.at(std::string{fwr_FMSGWR_SEC_predictor<INPUT,OUTPUT>::id_E});
339 auto Ws_new = W.at(std::string{fwr_FMSGWR_SEC_predictor<INPUT,OUTPUT>::id_S});
340 //input coherency
341 assert(We_new.size() == Ws_new.size());
342 for(std::size_t i = 0; i < We_new.size(); ++i){
343 assert((We_new[i].rows() == this->n_train()) && (We_new[i].cols() == this->n_train()));
344 assert((Ws_new[i].rows() == this->n_train()) && (Ws_new[i].cols() == this->n_train()));}
345 //number of units to be predicted
346 std::size_t n_pred = We_new.size();
347
348 //exact estimation
349 if(!this->in_cascade_estimation())
350 {
351 //compute the non-stationary betas in the new locations
352 //penalties in the new locations
353 //(j_tilde_tilde + Rs)^-1 (LsxLs)
354 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_double_tilde_Rs_inv = this->operator_comp().compute_penalty(m_psi_t,m_Xs_train_t,Ws_new,m_Xs_train,m_psi,m_Rs); //per applicarlo: j_double_tilde_RE_inv[i].solve(M) equivale a ([J_i_tilde_tilde + Re]^-1)*M
355 //(j_tilde + Re)^-1 (LexLe)
356 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_tilde_Re_inv = this->operator_comp().compute_penalty(m_Xe_train_crossed_t,We_new,m_Xe_train_crossed,m_Re);
357 //COMPUTING all the m_bs in the new locations, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATION-DEPENDENT BETAS (n_pred elements Lex1)
358 m_be_pred = this->operator_comp().compute_operator(m_Xe_train_crossed_t,We_new,m_y_tilde_new,j_tilde_Re_inv);
359 //COMPUTING all the m_be in the new locations, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATION-DEPENDENT BETAS (n_pred elements Lsx1)
360 m_bs_pred = this->operator_comp().compute_operator(m_psi_t,m_Xs_train_t,Ws_new,m_y_tilde_tilde_hat,j_double_tilde_Rs_inv);
361 }
362 //in cascade estimation
363 else
364 {
365 //be
366 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_i_Re_inv = this->operator_comp().compute_penalty(m_theta_t,m_Xe_train_t,We_new,m_Xe_train,m_theta,m_Re); //LexLe
367 m_be_pred = this->operator_comp().compute_operator(m_theta_t,m_Xe_train_t,We_new,m_y_tilde_hat,j_i_Re_inv); //(n_pred elements Lex1)
368 //bs
369 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_i_Rs_inv = this->operator_comp().compute_penalty(m_psi_t,m_Xs_train_t,Ws_new,m_Xs_train,m_psi,m_Rs); //LsxLs
370 m_bs_pred = this->operator_comp().compute_operator(m_psi_t,m_Xs_train_t,Ws_new,m_y_tilde_tilde_hat,j_i_Rs_inv); //(n_pred elements Lsx1)
371 }
372
373 //
374 //wrapping the b from the shape useful for the computation into a more useful format for reporting the results
375 //
376 //event-dependent covariates (qe elements of n_pred elements 1xLe_i)
377 m_Be_pred = this->operator_comp().wrap_operator(m_be_pred,m_Le_j,m_qe,n_pred);
378 //station-dependent covariates (qs elements of n_pred elements 1xLs_i)
379 m_Bs_pred = this->operator_comp().wrap_operator(m_bs_pred,m_Ls_j,m_qs,n_pred);
380 }

421 inline
422 functional_matrix<INPUT,OUTPUT>
423 predict(const std::map<std::string,functional_matrix<INPUT,OUTPUT>> &X_new)
424 const
425 override
426 {
427 //input coherency
428 assert(X_new.size() == 3);
429
430 auto Xc_new = X_new.at(std::string{fwr_FMSGWR_SEC_predictor<INPUT,OUTPUT>::id_C});
431 auto Xe_new = X_new.at(std::string{fwr_FMSGWR_SEC_predictor<INPUT,OUTPUT>::id_E});
432 auto Xs_new = X_new.at(std::string{fwr_FMSGWR_SEC_predictor<INPUT,OUTPUT>::id_S});
433
434 assert(Xc_new.rows() == Xe_new.rows());
435 assert(Xe_new.rows() == Xs_new.rows());
436 std::size_t n_pred = Xc_new.rows();
437 assert((n_pred == m_BetaE.size()) && (n_pred == m_BetaS.size()));
438 assert((Xc_new.cols() == m_qc) && (Xe_new.cols() == m_qe) && (Xs_new.cols() == m_qs));
439
440
441
442 //y_new = X_new*beta = Xc_new*beta_c + Xe_new*beta_e + Xs_new*beta_s
443 functional_matrix<INPUT,OUTPUT> y_new_C = fm_prod(Xc_new,m_BetaC,this->number_threads()); //n_pred x 1
444 functional_matrix<INPUT,OUTPUT> y_new_E(n_pred,1);
445 functional_matrix<INPUT,OUTPUT> y_new_S(n_pred,1);
446
447#ifdef _OPENMP
448#pragma omp parallel for shared(Xe_new,m_BetaE,y_new_E,Xs_new,m_BetaS,y_new_S,n_pred,m_qe,m_qs) num_threads(this->number_threads())
449#endif
450 for(std::size_t i = 0; i < n_pred; ++i)
451 {
452 std::vector< FUNC_OBJ<INPUT,OUTPUT> > xe_new_i(Xe_new.row(i).cbegin(),Xe_new.row(i).cend()); //1xqe
453 functional_matrix<INPUT,OUTPUT> Xe_new_i(xe_new_i,1,m_qe);
454 y_new_E(i,0) = fm_prod(Xe_new_i,m_BetaE[i],this->number_threads())(0,0);
455
456 std::vector< FUNC_OBJ<INPUT,OUTPUT> > xs_new_i(Xs_new.row(i).cbegin(),Xs_new.row(i).cend()); //1xqs
457 functional_matrix<INPUT,OUTPUT> Xs_new_i(xs_new_i,1,m_qs);
458 y_new_S(i,0) = fm_prod(Xs_new_i,m_BetaS[i],this->number_threads())(0,0);
459 }
460
461 return y_new_C + y_new_E + y_new_S;
462 }
