50template< FDAGWR_ALGO fdagwrType, typename INPUT = double, typename OUTPUT = double, class... Args >
51 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>)
52std::unique_ptr< fwr_predictor<INPUT,OUTPUT> >
53fwr_predictor_factory(Args &&... args)
54{
55 static_assert(fdagwrType == FDAGWR_ALGO::_FMSGWR_ESC_ ||
56 fdagwrType == FDAGWR_ALGO::_FMSGWR_SEC_ ||
57 fdagwrType == FDAGWR_ALGO::_FMGWR_ ||
58 fdagwrType == FDAGWR_ALGO::_FGWR_ ||
59 fdagwrType == FDAGWR_ALGO::_FWR_,
60 "Error in fdagwrType: wrong type specified.");
61
62 //predictor of FMSGWR_ESC
63 if constexpr (fdagwrType == FDAGWR_ALGO::_FMSGWR_ESC_)
64 return std::make_unique<fwr_FMSGWR_ESC_predictor<INPUT,OUTPUT>>(std::forward<Args>(args)...);
65
66 //predictor of FMSGWR_SEC
67 if constexpr (fdagwrType == FDAGWR_ALGO::_FMSGWR_SEC_)
68 return std::make_unique<fwr_FMSGWR_SEC_predictor<INPUT,OUTPUT>>(std::forward<Args>(args)...);
69
70 //predictor of FMGWR
71 if constexpr (fdagwrType == FDAGWR_ALGO::_FMGWR_)
72 return std::make_unique<fwr_FMGWR_predictor<INPUT,OUTPUT>>(std::forward<Args>(args)...);
73
74 //predictor of FGWR
75 if constexpr (fdagwrType == FDAGWR_ALGO::_FGWR_)
76 return std::make_unique<fwr_FGWR_predictor<INPUT,OUTPUT>>(std::forward<Args>(args)...);
77
78 //predictor of FWR
79 if constexpr (fdagwrType == FDAGWR_ALGO::_FWR_)
80 return std::make_unique<fwr_FWR_predictor<INPUT,OUTPUT>>(std::forward<Args>(args)...);
81}
