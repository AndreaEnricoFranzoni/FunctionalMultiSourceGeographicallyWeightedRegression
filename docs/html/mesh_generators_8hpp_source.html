98class Uniform : public OneDMeshGenerator
99{
100public:
106 Uniform(Geometry::Domain1D const &domain, unsigned int num_elements)
107 : OneDMeshGenerator(domain), M_num_elements(num_elements)
108 {}
109
114 MeshNodes operator()() const override
115 {
116 auto const &n = this->M_num_elements;
117 auto const &a = this->M_domain.left();
118 auto const &b = this->M_domain.right();
119 if(n == 0)
120 throw std::runtime_error("At least two elements");
121 MeshNodes mesh(n + 1);
122 double const h = (b - a) / static_cast<double>(n);
123#ifdef _OPENMP
124#pragma omp parallel for
125#endif
126 for(auto i = 0u; i < n; ++i)
127 mesh[i] = a + h * i;
128
129 mesh[n] = b;
130 return mesh;
131 }
132
133private:
135 std::size_t M_num_elements;
136};
