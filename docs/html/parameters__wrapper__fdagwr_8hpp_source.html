49template < FDAGWR_COVARIATES_TYPES fdagwr_cov_t >
50inline
51void
52check_dim_input(std::size_t sz_q,
53 std::size_t sz_input,
54 const std::string& quantity)
55{
56 // throwing an exception if dimensions not coinciding
57 if(sz_q!=sz_input){
58 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
59 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
60 std::string error_message = "For " + covariates_type + " covariates, " + quantity + " size has to be " + std::to_string(sz_q);
61 throw std::invalid_argument(error_message);}
62}

74template < FDAGWR_COVARIATES_TYPES fdagwr_cov_t >
75inline
76std::vector<std::string>
77wrap_covariates_names(Rcpp::List cov_coeff_list)
78{
79 // name of the covariates in the input list list
80 Rcpp::Nullable<Rcpp::CharacterVector> cov_names_input_list = cov_coeff_list.names();
81 // number of covariates
82 std::size_t number_cov = cov_coeff_list.size();
83 // type of the covariates
84 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
85
86 //if all the covariates do not own a name: put default names for all of them
87 if (cov_names_input_list.isNull()){
88 //output container
89 std::vector<std::string> covariates_names;
90 covariates_names.reserve(number_cov);
91 //put a default value
92 for(std::size_t i = 0; i < number_cov; ++i){ covariates_names.emplace_back("Covariate" + covariates_type + std::to_string(i+1));}
93
94 return covariates_names;}
95
96 //copy the actual names
97 std::vector<std::string> covariates_names = as<std::vector<std::string>>(cov_names_input_list);
98 //defaulting only missing names
99 for(std::size_t i = 0; i < number_cov; ++i){
100 if(covariates_names[i] == "" ){
101 covariates_names[i] = "Covariate" + covariates_type + std::to_string(i+1);}}
102
103 return covariates_names;
104}

118inline
119std::vector<FDAGWR_TRAITS::Dense_Matrix>
120wrap_covariates_coefficients(Rcpp::List cov_coeff_list)
121{
122 // number of covariates
123 std::size_t number_cov = cov_coeff_list.size();
124
125 //where to store the covariates coefficients
126 std::vector<FDAGWR_TRAITS::Dense_Matrix> covariates_coefficients;
127 covariates_coefficients.reserve(number_cov);
128
129 //check and read the coefficients for all the covariates
130 for(std::size_t i = 0; i < number_cov; ++i){
131
132 //read the data
133 covariates_coefficients.push_back(reader_data<double,REM_NAN::MR>(cov_coeff_list[i]));
134
135 //checking that all the coefficients refer to the same amount of statistical units (checking that all the list elements have the same number of columns)
136 if(i>0 && covariates_coefficients[i-1].cols()!=covariates_coefficients[i].cols()){
137 std::string error_message = "All covariates coefficients have to refer to the same number of statistical units";
138 throw std::invalid_argument(error_message);}
139 }
140
141 return covariates_coefficients;
142}

154inline
155std::string
156wrap_and_check_basis_type(const std::string &basis_types_name)
157{
158 //checking if the requested basis type is implemented
159 if (std::find(FDAGWR_BASIS_TYPES::_implemented_basis_.cbegin(),FDAGWR_BASIS_TYPES::_implemented_basis_.cend(),basis_types_name) == FDAGWR_BASIS_TYPES::_implemented_basis_.cend()){
160 //throwing an exception if not
161 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
162 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
163 std::string error_message = "For " + covariates_type + ", basis type " + basis_types_name + " is not acceptable: basis types accepted: ";
164 for(auto it = FDAGWR_BASIS_TYPES::_implemented_basis_.cbegin(); it != FDAGWR_BASIS_TYPES::_implemented_basis_.cend(); ++it){
165 error_message += *it;
166 if (next(it)!=FDAGWR_BASIS_TYPES::_implemented_basis_.cend()){
167 error_message += ", ";}}
168 throw std::invalid_argument(error_message);}
169
170 return basis_types_name;
171}

189 std::size_t number_of_covariates)
190{
191 //default if a NULL is passed
192 if (basis_types_names.isNull()){
193 //returning all "bsplines"
194 std::vector<std::string> basis_types_names_wrapped(number_of_covariates,std::string{FDAGWR_BASIS_TYPES::_bsplines_});
195 return basis_types_names_wrapped;}
196
197 //wrapper
198 std::vector<std::string> basis_types_names_wrapped = Rcpp::as<std::vector<std::string>>(basis_types_names);
199 // number of covariates in the passed input
200 std::size_t q = basis_types_names_wrapped.size();
201 //checking if correct dimension of the passed input
202 check_dim_input<fdagwr_cov_t>(number_of_covariates,q,"vector with the basis types");
203
204 //checking if the requested basis type is implemented, for all the covariates
205 for(std::size_t i = 0; i < q; ++i){
206 //throwing an exception if not
207 if (std::find(FDAGWR_BASIS_TYPES::_implemented_basis_.cbegin(),FDAGWR_BASIS_TYPES::_implemented_basis_.cend(),basis_types_names_wrapped[i]) == FDAGWR_BASIS_TYPES::_implemented_basis_.cend()){
208 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
209 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
210 std::string error_message = "For " + covariates_type + " covariates, basis type " + basis_types_names_wrapped[i] + " is not acceptable: basis types accepted: ";
211 for(auto it = FDAGWR_BASIS_TYPES::_implemented_basis_.cbegin(); it != FDAGWR_BASIS_TYPES::_implemented_basis_.cend(); ++it){
212 error_message += *it;
213 if (next(it)!=FDAGWR_BASIS_TYPES::_implemented_basis_.cend()){
214 error_message += ", ";}}
215 throw std::invalid_argument(error_message);}
216 }
217
218 return basis_types_names_wrapped;
219}

247inline
248std::vector<double>
249wrap_abscissas(Rcpp::NumericVector abscissas,
250 double a,
251 double b)
252{
253 //check that domain extremes are consistent
254 if(a>=b){
255 //throwing an exception if not
256 std::string error_message1 = "Left extreme of the domain has to be lower than the right one";
257 throw std::invalid_argument(error_message1);}
258
259 //sorting the abscissas values (security check)
260 std::vector<double> abscissas_wrapped = Rcpp::as<std::vector<double>>(abscissas);
261 std::sort(abscissas_wrapped.begin(),abscissas_wrapped.end());
262
263 //checking that the passed points are inside the domain
264 if(abscissas_wrapped[0] < a || abscissas_wrapped.back() > b){
265 //throwing an exception if not
266 std::string error_message2 = "The points in which there are the discrete evaluations of the curves have to be in the interval (" + std::to_string(a) + "," + std::to_string(b) + ")";
267 throw std::invalid_argument(error_message2);}
268
269 return abscissas_wrapped;
270}

287template < FDAGWR_COVARIATES_TYPES fdagwr_cov_t >
288inline
289std::map<std::string,std::size_t>
290wrap_and_check_basis_number_and_degree(Rcpp::Nullable<int> basis_number,
291 Rcpp::Nullable<int> basis_degree,
292 std::size_t knots_size,
293 std::string basis_type)
294{
295 //container to store the returning element
296 std::map<std::string,std::size_t> returning_element;
297
298
299 //"constant" basis type case
300 if(basis_type == std::string{FDAGWR_BASIS_TYPES::_constant_})
301 {
302 //default (and only possible) case
303 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::n_basis_string},constant_basis<FDAGWR_TRAITS::basis_geometry>::number_of_basis_constant_basis)); //one base
304 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::degree_basis_string},constant_basis<FDAGWR_TRAITS::basis_geometry>::degree_constant_basis)); //degree zero
305
306 return returning_element;
307 }
308
309
310 //"bsplines" basis type case
311
312 //basis number unknown, order unknown: default values
313 if(basis_number.isNull() && basis_degree.isNull())
314 {
315 std::size_t degree = bsplines_basis<FDAGWR_TRAITS::basis_geometry>::bsplines_degree_default; //default is a cubic B-spline (degree 3)
316 std::size_t n_basis = degree + knots_size - static_cast<std::size_t>(1); //B-splines constraint for basis number
317
318 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::n_basis_string},n_basis));
319 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::degree_basis_string},degree));
320 }
321
322 //basis number unknown, degree known
323 if (basis_number.isNull() && basis_degree.isNotNull())
324 {
325 //throwing an exception if a negative degree is passed
326 if (Rcpp::as<int>(basis_degree) < 0){
327 std::string error_message1 = "Basis degree for the response has to be a non-negative integer";
328 throw std::invalid_argument(error_message1);}
329
330 //computing the basis number
331 std::size_t n_basis = static_cast<std::size_t>(Rcpp::as<int>(basis_degree)) + knots_size - static_cast<std::size_t>(1);
332
333 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::n_basis_string},n_basis));
334 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::degree_basis_string},static_cast<std::size_t>(Rcpp::as<int>(basis_degree))));
335 }
336
337 //basis number known, degree unknown
338 if (basis_number.isNotNull() && basis_degree.isNull())
339 {
340 //throwing an exception if basis_number < knots_size - 1, that would inquire a negative degree
341 if (static_cast<std::size_t>(Rcpp::as<int>(basis_number)) < knots_size - static_cast<std::size_t>(1)){
342 std::string error_message2 = "The number of basis for the response has to be at least the number of knots (" + std::to_string(knots_size) + ") - 1";
343 throw std::invalid_argument(error_message2);}
344
345 //computing the degree
346 std::size_t degree = static_cast<std::size_t>(Rcpp::as<int>(basis_number)) - knots_size + static_cast<std::size_t>(1);
347
348 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::n_basis_string},static_cast<std::size_t>(Rcpp::as<int>(basis_number))));
349 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::degree_basis_string},degree));
350 }
351
352 //both basis number and order known
353 if (basis_number.isNotNull() && basis_degree.isNotNull())
354 {
355 //thrwoing an exception if a negative basis degree is passed
356 if (Rcpp::as<int>(basis_degree) < 0){
357 std::string error_message3 = "Basis degree for the response has to be a non-negative integer";
358 throw std::invalid_argument(error_message3);}
359 //thrwoing an exception if a non-positive basis number is passed
360 if (Rcpp::as<int>(basis_number) <= 0){
361 std::string error_message4 = "Basis number for the response has to be a positive integer";
362 throw std::invalid_argument(error_message4);}
363 //throwing an exception if basis_number != basis_degree + knots_size - 1
364 if (static_cast<std::size_t>(Rcpp::as<int>(basis_number)) != static_cast<std::size_t>(Rcpp::as<int>(basis_degree)) + knots_size - static_cast<std::size_t>(1)){
365 std::string error_message5 = "The number of basis for the response has to be the order of the basis (" + std::to_string(static_cast<std::size_t>(Rcpp::as<int>(basis_degree))) + ") + the number of knots (" + std::to_string(knots_size) + ") - 1";
366 throw std::invalid_argument(error_message5);}
367
368 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::n_basis_string},static_cast<std::size_t>(static_cast<std::size_t>(Rcpp::as<int>(basis_number)))));
369 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::degree_basis_string},static_cast<std::size_t>(static_cast<std::size_t>(Rcpp::as<int>(basis_degree)))));
370 }
371
372 return returning_element;
373}

391template < FDAGWR_COVARIATES_TYPES fdagwr_cov_t >
392inline
393std::map<std::string,std::vector<std::size_t>>
394wrap_and_check_basis_number_and_degree(Rcpp::Nullable<Rcpp::IntegerVector> basis_numbers,
395 Rcpp::Nullable<Rcpp::IntegerVector> basis_degrees,
396 std::size_t knots_size,
397 std::size_t number_of_covariates,
398 const std::vector<std::string> & basis_types)
399{
400 //container to store the returning element
401 std::map<std::string,std::vector<std::size_t>> returning_element;
402
403 //basis number unknown, order unknown: default values
404 if (basis_numbers.isNull() && basis_degrees.isNull())
405 {
406 std::vector<std::size_t> degrees(number_of_covariates,bsplines_basis<FDAGWR_TRAITS::basis_geometry>::bsplines_degree_default); //default is a cubic B-spline (degree 3) for all the covariates
407 std::vector<std::size_t> ns_basis(number_of_covariates,bsplines_basis<FDAGWR_TRAITS::basis_geometry>::bsplines_degree_default + knots_size - static_cast<std::size_t>(1));
408
409 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::n_basis_string},ns_basis));
410 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::degree_basis_string},degrees));
411 }
412
413 //basis number unknown, order known
414 if (basis_numbers.isNull() && basis_degrees.isNotNull())
415 {
416 //wraping the R input
417 auto basis_degrees_w = Rcpp::as<std::vector<int>>(basis_degrees);
418
419 //check the correct dimension of the input (number of covariates, indeed), if not throwing an exception
420 if (basis_degrees_w.size() != number_of_covariates){
421 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
422 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
423 std::string error_message1 = "It is necessary to pass a vector with " + std::to_string(number_of_covariates) + " basis degrees for the " + covariates_type + " covariates";
424 throw std::invalid_argument(error_message1);}
425
426 //checking that the input is consistent (basis degrees non-negative for all the covariates), if not throwing an exception
427 auto min_basis_degree = std::min_element(basis_degrees_w.cbegin(),basis_degrees_w.cend());
428 if (*min_basis_degree < 0){
429 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
430 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
431 std::string error_message2 = "Basis degrees for all the " + covariates_type + " covariates have to be non-negative";
432 throw std::invalid_argument(error_message2);}
433
434 //converting into a vector of std::size_t (create another copy for the conversion, they are objects of small dimensions)
435 std::vector<std::size_t> degrees;
436 degrees.resize(number_of_covariates);
437 std::transform(basis_degrees_w.cbegin(),
438 basis_degrees_w.cend(),
439 degrees.begin(),
440 [](auto el){return static_cast<std::size_t>(el);});
441
442 //computing the number of basis for each covariate accordingly
443 std::vector<std::size_t> ns_basis;
444 ns_basis.resize(number_of_covariates);
445 std::transform(degrees.cbegin(),
446 degrees.cend(),
447 ns_basis.begin(),
448 [knots_size](std::size_t const &el){ return (el + knots_size - static_cast<std::size_t>(1));});
449
450 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::n_basis_string},ns_basis));
451 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::degree_basis_string},degrees));
452 }
453
454 //basis number known, order unknown
455 if (basis_numbers.isNotNull() && basis_degrees.isNull())
456 {
457 //wrapping the R input
458 auto basis_numbers_w = Rcpp::as<std::vector<int>>(basis_numbers);
459
460 //check the correct dimension of the input (number of covariates, indeed), if not throwing an exception
461 if (basis_numbers_w.size() != number_of_covariates){
462 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
463 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
464 std::string error_message3 = "It is necessary to pass a vector with " + std::to_string(number_of_covariates) + " number of basis for the " + covariates_type + " covariates";
465 throw std::invalid_argument(error_message3);}
466
467 //checking that the input is consistent (basis number at least the number of knots - 1 for all the covariates for bsplines), if not throwing an exception
468 for(std::size_t i = 0; i < number_of_covariates; ++i){
469 if ((basis_numbers_w[i] < (knots_size - static_cast<std::size_t>(1))) && basis_types[i] == FDAGWR_BASIS_TYPES::_bsplines_){ //checking B-splines relationship for each covariate, where basis are bsplines
470 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
471 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
472 std::string error_message4 = "The number of basis, bspline basis, for all the " + covariates_type + " covariates has to be at least the number of knots (" + std::to_string(knots_size) + ") - 1";
473 throw std::invalid_argument(error_message4);}}
474
475 //converting into a vector of std::size_t (create another copy for the conversion, they are objects of small dimensions)
476 std::vector<std::size_t> ns_basis;
477 ns_basis.resize(number_of_covariates);
478 std::transform(basis_numbers_w.cbegin(),
479 basis_numbers_w.cend(),
480 ns_basis.begin(),
481 [](auto el){return static_cast<std::size_t>(el);});
482
483 //computing the order of the basis for each covariate
484 std::vector<std::size_t> degrees;
485 degrees.resize(number_of_covariates);
486 std::transform(ns_basis.cbegin(),
487 ns_basis.cend(),
488 degrees.begin(),
489 [knots_size](std::size_t const &el){return (el - knots_size + static_cast<std::size_t>(1));});
490
491 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::n_basis_string},ns_basis));
492 returning_element.insert(std::make_pair(std::string{FDAGWR_FEATS::degree_basis_string},degrees));
493 }
494
495 //both basis number and order known
496 if (basis_numbers.isNotNull() && basis_degrees.isNotNull())
497 {
498 //wrapping the R input
499 auto basis_degrees_w = Rcpp::as<std::vector<int>>(basis_degrees);
500 auto basis_numbers_w = Rcpp::as<std::vector<int>>(basis_numbers);
501
502 //check the correct dimension of the inputs (number of covariates, indeed), if not throwing an exception
503 if (basis_degrees_w.size() != number_of_covariates){
504 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
505 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
506 std::string error_message5 = "It is necessary to pass a vector with " + std::to_string(number_of_covariates) + " basis degrees for the " + covariates_type + " covariates";
507 throw std::invalid_argument(error_message5);}
508 if (basis_numbers_w.size() != number_of_covariates){
509 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
510 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
511 std::string error_message6 = "It is necessary to pass a vector with " + std::to_string(number_of_covariates) + " number of basis for the " + covariates_type + " covariates";
512 throw std::invalid_argument(error_message6);}
513
514 //checking that the degrees of basis input are consistent (basis order non-negative for all the covariates), if not throwing an exception
515 auto min_basis_degree = std::min_element(basis_degrees_w.cbegin(),basis_degrees_w.cend());
516 if (*min_basis_degree < 0){
517 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
518 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
519 std::string error_message7 = "Basis degrees for all the " + covariates_type + " covariates have to be non-negative";
520 throw std::invalid_argument(error_message7);}
521
522 //converting into a vector of std::size_t (create another copy for the conversion, they are objects of small dimensions)
523 //numbers of basis
524 std::vector<std::size_t> ns_basis;
525 ns_basis.resize(number_of_covariates);
526 std::transform(basis_numbers_w.cbegin(),
527 basis_numbers_w.cend(),
528 ns_basis.begin(),
529 [](auto el){return static_cast<std::size_t>(el);});
530 //degrees of basis
531 std::vector<std::size_t> degrees;
532 degrees.resize(number_of_covariates);
533 std::transform(basis_degrees_w.cbegin(),
534 basis_degrees_w.cend(),
535 degrees.begin(),
536 [](auto el){return static_cast<std::size_t>(el);});
537
538 //check that the number of basis input is consistent, if not throwing an exception
539 for(std::size_t i = 0; i < number_of_covariates; ++i){
540 //checking B-splines relationship for each covariate, where basis are bsplines
541 if ((ns_basis[i] - (degrees[i] + knots_size - static_cast<std::size_t>(1)) != 0) && basis_types[i] == FDAGWR_BASIS_TYPES::_bsplines_){
542 std::string covariates_type = std::string{covariate_type<fdagwr_cov_t>()};
543 std::transform(covariates_type.begin(),covariates_type.end(),covariates_type.begin(),[](unsigned char c) { return std::tolower(c);});
544 std::string error_message8 = "The number of basis, bspline basis, for the " + covariates_type + " covariates has to be the degree of the basis + the number of knots - 1";
545 throw std::invalid_argument(error_message8);}}

549 }
550

560 }

563}
