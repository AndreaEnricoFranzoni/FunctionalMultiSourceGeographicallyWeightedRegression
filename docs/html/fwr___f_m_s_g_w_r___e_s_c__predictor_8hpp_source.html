285 inline
286 void
287 computePartialResiduals()
288 override
289 {
290 //exact estimation
291 if(!this->in_cascade_estimation())
292 {
293 //retrieve the partial residuals from the fitted model
294 //K_e_s(t) (n_train x Ls)
295 m_K_e_s = this->operator_comp().compute_functional_operator(m_Xe_train,m_theta,m_B_e_for_K_e_s);
296 //X_s_crossed(t) (n_train x Ls)
297 m_X_s_train_crossed = fm_prod(m_Xs_train,m_psi) - m_K_e_s;
298 m_X_s_train_crossed_t = m_X_s_train_crossed.transpose();
299 //y_tilde_hat(t) (n_trainx1)
300 m_y_tilde_hat = fm_prod(m_phi,m_c_tilde_hat);
301 //He(t) (n_trainx(n_train*Ly))
302 m_H_e = this->operator_comp().compute_functional_operator(m_Xe_train,m_theta,m_A_e);
303 //y_tilde_new(t) (n_trainx1)
304 m_y_tilde_new = fm_prod(functional_matrix<INPUT,OUTPUT>(m_phi - m_H_e),m_c_tilde_hat,this->number_threads());
305 //y_tilde_tilde_hat(t) (n_trainx1)
306 m_y_tilde_tilde_hat = m_y_tilde_hat - this->operator_comp().compute_functional_operator(m_Xs_train,m_psi,m_bs_fitted);
307 }
308 //in cascade estimation
309 else
310 {
311 //y_tilde_hat(t) (n_trainx1)
312 m_y_tilde_hat = m_y_train - fm_prod(fm_prod(m_Xc_train,m_omega),m_bc_fitted,this->number_threads());
313 //y_tilde_tilde_hat(t) (n_trainx1)
314 m_y_tilde_tilde_hat = m_y_train;
315
316#ifdef _OPENMP
317#pragma omp parallel for shared(m_y_tilde_tilde_hat,m_Xs_train,m_psi,m_y_tilde_hat) num_threads(this->number_threads())
318#endif
319 for(std::size_t i = 0; i < this->n_train(); ++i)
320 {
321 std::vector< FUNC_OBJ<INPUT,OUTPUT> > xs_i(m_Xs_train.row(i).cbegin(),m_Xs_train.row(i).cend()); //1xqs
322 functional_matrix<INPUT,OUTPUT> Xs_i(xs_i,1,m_qs);
323 functional_matrix<INPUT,OUTPUT> y_tilde_hat_i(1,1,m_y_tilde_hat(i,0));
324 m_y_tilde_tilde_hat(i,0) = (y_tilde_hat_i - fm_prod(fm_prod(Xs_i,m_psi),m_bs_fitted[i],this->number_threads()))(0,0);
325 }
326 }
327 }

335 inline
336 void
337 computeBNew(const std::map<std::string,std::vector< functional_matrix_diagonal<INPUT,OUTPUT> >> &W)
338 override
339 {
340 assert(W.size() == 2);
341 auto We_new = W.at(std::string{fwr_FMSGWR_ESC_predictor<INPUT,OUTPUT>::id_E});
342 auto Ws_new = W.at(std::string{fwr_FMSGWR_ESC_predictor<INPUT,OUTPUT>::id_S});
343 //input coherency
344 assert(We_new.size() == Ws_new.size());
345 for(std::size_t i = 0; i < We_new.size(); ++i){
346 assert((We_new[i].rows() == this->n_train()) && (We_new[i].cols() == this->n_train()));
347 assert((Ws_new[i].rows() == this->n_train()) && (Ws_new[i].cols() == this->n_train()));}
348 //number of units to be predicted
349 std::size_t n_pred = We_new.size();
350
351 //exact estimation
352 if(!this->in_cascade_estimation())
353 {
354 //compute the non-stationary betas in the new locations
355 //penalties in the new locations
356 //(j_tilde_tilde + Re)^-1
357 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_double_tilde_Re_inv = this->operator_comp().compute_penalty(m_theta_t,m_Xe_train_t,We_new,m_Xe_train,m_theta,m_Re); //per applicarlo: j_double_tilde_RE_inv[i].solve(M) equivale a ([J_i_tilde_tilde + Re]^-1)*M
358 //(j_tilde + Rs)^-1
359 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_tilde_Rs_inv = this->operator_comp().compute_penalty(m_X_s_train_crossed_t,Ws_new,m_X_s_train_crossed,m_Rs);
360 //COMPUTING all the m_bs in the new locations, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATION-DEPENDENT BETAS (n_pred elements Lsx1)
361 m_bs_pred = this->operator_comp().compute_operator(m_X_s_train_crossed_t,Ws_new,m_y_tilde_new,j_tilde_Rs_inv);
362 //COMPUTING all the m_be in the new locations, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE STATION-DEPENDENT BETAS (n_pred elements Lex1)
363 m_be_pred = this->operator_comp().compute_operator(m_theta_t,m_Xe_train_t,We_new,m_y_tilde_tilde_hat,j_double_tilde_Re_inv);
364 }
365 //in cascade estimation
366 else
367 {
368 //bs
369 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_i_Rs_inv = this->operator_comp().compute_penalty(m_psi_t,m_Xs_train_t,Ws_new,m_Xs_train,m_psi,m_Rs); //LsxLs
370 m_bs_pred = this->operator_comp().compute_operator(m_psi_t,m_Xs_train_t,Ws_new,m_y_tilde_hat,j_i_Rs_inv); //(n_pred elements Lsx1)
371 //be
372 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_i_Re_inv = this->operator_comp().compute_penalty(m_theta_t,m_Xe_train_t,We_new,m_Xe_train,m_theta,m_Re); //LexLe
373 m_be_pred = this->operator_comp().compute_operator(m_theta_t,m_Xe_train_t,We_new,m_y_tilde_tilde_hat,j_i_Re_inv); //(n_pred elements Lex1)
374 }
375
376 //
377 //wrapping the b from the shape useful for the computation into a more useful format for reporting the results
378 //
379 //event-dependent covariates (qe elements of n_pred elements 1xLe_i)
380 m_Be_pred = this->operator_comp().wrap_operator(m_be_pred,m_Le_j,m_qe,n_pred);
381 //station-dependent covariates (qs elements of n_pred elements 1xLs_i)
382 m_Bs_pred = this->operator_comp().wrap_operator(m_bs_pred,m_Ls_j,m_qs,n_pred);
383
384 }

425 inline
426 functional_matrix<INPUT,OUTPUT>
427 predict(const std::map<std::string,functional_matrix<INPUT,OUTPUT>> &X_new)
428 const
429 override
430 {
431 //input coherency
432 assert(X_new.size() == 3);
433
434 auto Xc_new = X_new.at(std::string{fwr_FMSGWR_ESC_predictor<INPUT,OUTPUT>::id_C});
435 auto Xe_new = X_new.at(std::string{fwr_FMSGWR_ESC_predictor<INPUT,OUTPUT>::id_E});
436 auto Xs_new = X_new.at(std::string{fwr_FMSGWR_ESC_predictor<INPUT,OUTPUT>::id_S});
437
438 assert(Xc_new.rows() == Xe_new.rows());
439 assert(Xe_new.rows() == Xs_new.rows());
440 std::size_t n_pred = Xc_new.rows();
441 assert((n_pred == m_BetaE.size()) && (n_pred == m_BetaS.size()));
442 assert((Xc_new.cols() == m_qc) && (Xe_new.cols() == m_qe) && (Xs_new.cols() == m_qs));
443
444
445
446 //y_new = X_new*beta = Xc_new*beta_c + Xe_new*beta_e + Xs_new*beta_s
447 functional_matrix<INPUT,OUTPUT> y_new_C = fm_prod(Xc_new,m_BetaC,this->number_threads()); //n_pred x 1
448 functional_matrix<INPUT,OUTPUT> y_new_E(n_pred,1);
449 functional_matrix<INPUT,OUTPUT> y_new_S(n_pred,1);
450
451#ifdef _OPENMP
452#pragma omp parallel for shared(Xe_new,m_BetaE,y_new_E,Xs_new,m_BetaS,y_new_S,n_pred,m_qe,m_qs) num_threads(this->number_threads())
453#endif
454 for(std::size_t i = 0; i < n_pred; ++i)
455 {
456 std::vector< FUNC_OBJ<INPUT,OUTPUT> > xe_new_i(Xe_new.row(i).cbegin(),Xe_new.row(i).cend()); //1xqe
457 functional_matrix<INPUT,OUTPUT> Xe_new_i(xe_new_i,1,m_qe);
458 y_new_E(i,0) = fm_prod(Xe_new_i,m_BetaE[i],this->number_threads())(0,0);
459
460 std::vector< FUNC_OBJ<INPUT,OUTPUT> > xs_new_i(Xs_new.row(i).cbegin(),Xs_new.row(i).cend()); //1xqs
461 functional_matrix<INPUT,OUTPUT> Xs_new_i(xs_new_i,1,m_qs);
462 y_new_S(i,0) = fm_prod(Xs_new_i,m_BetaS[i],this->number_threads())(0,0);
463 }
464
465 return y_new_C + y_new_E + y_new_S;
466 }
