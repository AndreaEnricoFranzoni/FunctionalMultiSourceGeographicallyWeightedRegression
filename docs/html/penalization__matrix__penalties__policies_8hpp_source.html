38struct SecondDerivativePenalty
39{
45 Eigen::SparseMatrix<double>
46 operator()(const basis_systems< FDAGWR_TRAITS::basis_geometry, bsplines_basis > &bs, std::size_t system_number)
47 const
48 {
49 //using fdaPDE
50 fdapde::TrialFunction u(bs.systems_of_basis()[system_number].basis());
51 fdapde::TestFunction v(bs.systems_of_basis()[system_number].basis());
52 // stiff matrix: penalizing the second derivaive
53 auto stiff = integral(bs.knots())(dxx(u) * dxx(v));
54
55 //assmebling the stiff matrix
56 return stiff.assemble();
57 }
58};
