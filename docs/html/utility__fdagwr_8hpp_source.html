107Rcpp::List
108toRList(const std::vector< FDAGWR_TRAITS::Dense_Matrix >& b,
109 bool add_unit_number = false)
110{
111 //WRAP B STATIONARY
112 Rcpp::List out(b.size());
113 Rcpp::CharacterVector names(b.size());
114 for (std::size_t i = 0; i < b.size(); i++) {
115 out[i] = Rcpp::wrap(b[i]); // RcppEigen converts Eigen::MatrixXd
116 names[i] = std::string("unit_") + std::to_string(i+1);
117 }
118
119 if (add_unit_number){ out.names() = names;}
120
121 return out;
122}

181Rcpp::List
182toRList(const std::vector< std::vector< FDAGWR_TRAITS::Dense_Matrix >>& b)
183{
184 // WRAP B NON-STATIONARY
185 //for each non-stationary covariate
186 Rcpp::List outerList(b.size());
187
188 for(std::size_t i = 0; i < b.size(); ++i){
189 Rcpp::List innerList(b[i].size());
190 Rcpp::CharacterVector innerNames(b[i].size());
191 //for each unit
192 for(std::size_t j = 0; j < b[i].size(); ++j){
193 innerList[j] = Rcpp::wrap(b[i][j]);
194 innerNames[j] = std::string("unit_") + std::to_string(j+1);
195 }
196
197 innerList.names() = innerNames;
198 outerList[i]=(innerList);
199 }
200 return outerList;
201}

383Rcpp::List
384wrap_b_to_R_list(const BTuple& b,
385 const std::vector<std::string>& names_bc = {},
386 const std::vector<std::string>& basis_type_bc = {},
387 const std::vector<std::size_t>& basis_number_bc = {},
388 const std::vector<FDAGWR_TRAITS::fd_obj_x_type> & knots_bc = {},
389 const std::vector<std::string>& names_bnc = {},
390 const std::vector<std::string>& basis_type_bnc = {},
391 const std::vector<std::size_t>& basis_number_bnc = {},
392 const std::vector<FDAGWR_TRAITS::fd_obj_x_type> & knots_bnc = {},
393 const std::vector<std::string>& names_be = {},
394 const std::vector<std::string>& basis_type_be = {},
395 const std::vector<std::size_t>& basis_number_be = {},
396 const std::vector<FDAGWR_TRAITS::fd_obj_x_type> & knots_be = {},
397 const std::vector<std::string>& names_bs = {},
398 const std::vector<std::string>& basis_type_bs = {},
399 const std::vector<std::size_t>& basis_number_bs = {},
400 const std::vector<FDAGWR_TRAITS::fd_obj_x_type> & knots_bs = {})
401{
402 //using std::visit to wrap the betas basis expansion depending on the type of model fitted
403 return std::visit([&](auto&& tup) -> Rcpp::List {
404 using T = std::decay_t<decltype(tup)>;
405
406 //FWR
407 if constexpr (std::is_same_v<T, std::tuple< std::vector< FDAGWR_TRAITS::Dense_Matrix >>>)
408 {
409 //stationary b
410 Rcpp::List bc = toRList(std::get<0>(tup),basis_type_bc,basis_number_bc,knots_bc);
411 if (!names_bc.empty())
412 bc.names() = Rcpp::CharacterVector(names_bc.cbegin(), names_bc.cend());
413
414 return Rcpp::List::create(Rcpp::Named(std::string{FDAGWR_B_NAMES::bc}) = bc);
415 }
416
417 //FGWR
418 if constexpr (std::is_same_v<T, std::tuple< std::vector< std::vector< FDAGWR_TRAITS::Dense_Matrix >>>>)
419 {
420 //non stationary b
421 Rcpp::List bnc = toRList(std::get<0>(tup),basis_type_bnc,basis_number_bnc,knots_bnc);
422 if(!names_bnc.empty())
423 bnc.names() = Rcpp::CharacterVector(names_bnc.cbegin(), names_bnc.cend());
424
425 return Rcpp::List::create(Rcpp::Named(std::string{FDAGWR_B_NAMES::bnc}) = bnc);
426 }
427
428 //FMGWR
429 else if constexpr (std::is_same_v<T, std::tuple< std::vector< FDAGWR_TRAITS::Dense_Matrix >,
430 std::vector< std::vector< FDAGWR_TRAITS::Dense_Matrix >>>>)
431 {
432 //stationary b
433 Rcpp::List bc = toRList(std::get<0>(tup),basis_type_bc,basis_number_bc,knots_bc);
434 if (!names_bc.empty())
435 bc.names() = Rcpp::CharacterVector(names_bc.cbegin(), names_bc.cend());
436 //non-stationary b
437 Rcpp::List bnc = toRList(std::get<1>(tup),basis_type_bnc,basis_number_bnc,knots_bnc);
438 if (!names_bnc.empty())
439 bnc.names() = Rcpp::CharacterVector(names_bnc.cbegin(), names_bnc.cend());
440
441 return Rcpp::List::create(Rcpp::Named(std::string{FDAGWR_B_NAMES::bc}) = bc,
442 Rcpp::Named(std::string{FDAGWR_B_NAMES::bnc}) = bnc);
443 }
444
445 //FMSGWR
446 else if constexpr (std::is_same_v<T, std::tuple< std::vector< FDAGWR_TRAITS::Dense_Matrix >,
447 std::vector< std::vector< FDAGWR_TRAITS::Dense_Matrix >>,
448 std::vector< std::vector< FDAGWR_TRAITS::Dense_Matrix >>>>)
449 {
450 //stationary b
451 Rcpp::List bc = toRList(std::get<0>(tup),basis_type_bc,basis_number_bc,knots_bc);
452 if (!names_bc.empty())
453 bc.names() = Rcpp::CharacterVector(names_bc.cbegin(), names_bc.cend());
454 //event b
455 Rcpp::List be = toRList(std::get<1>(tup),basis_type_be,basis_number_be,knots_be);
456 if (!names_be.empty())
457 be.names() = Rcpp::CharacterVector(names_be.cbegin(), names_be.cend());
458 //station b
459 Rcpp::List bs = toRList(std::get<2>(tup),basis_type_bs,basis_number_bs,knots_bs);
460 if (!names_bs.empty())
461 bs.names() = Rcpp::CharacterVector(names_bs.cbegin(), names_bs.cend());
462
463 return Rcpp::List::create(Rcpp::Named(std::string{FDAGWR_B_NAMES::bc}) = bc,
464 Rcpp::Named(std::string{FDAGWR_B_NAMES::be}) = be,
465 Rcpp::Named(std::string{FDAGWR_B_NAMES::bs}) = bs);
466 }
467 }, b);
468}

565Rcpp::List
566wrap_PRes_to_R_list(const PartialResidualTuple& p_res)
567{
568 //using std::visit to wrap the element needed to reconstruct the partial residuals of a specific fwr model
569 return std::visit([&](auto&& tup) -> Rcpp::List {
570 using T = std::decay_t<decltype(tup)>;
571
572 //FMSGWR
573 if constexpr (std::is_same_v<T, std::tuple< FDAGWR_TRAITS::Dense_Matrix, std::vector< FDAGWR_TRAITS::Dense_Matrix >, std::vector< FDAGWR_TRAITS::Dense_Matrix >>>)
574 {
575 return Rcpp::List::create(Rcpp::Named(std::string{FDAGWR_HELPERS_for_PRED_NAMES::p_res_c_tilde_hat}) = Rcpp::wrap(std::get<0>(tup)),
576 Rcpp::Named(std::string{FDAGWR_HELPERS_for_PRED_NAMES::p_res_A__}) = toRList(std::get<1>(tup),false),
577 Rcpp::Named(std::string{FDAGWR_HELPERS_for_PRED_NAMES::p_res_B__for_K}) = toRList(std::get<2>(tup),false));
578 }
579
580 //FMGWR
581 else if constexpr (std::is_same_v<T, std::tuple< FDAGWR_TRAITS::Dense_Matrix >>)
582 {
583 return Rcpp::List::create(Rcpp::Named(std::string{FDAGWR_HELPERS_for_PRED_NAMES::p_res_c_tilde_hat}) = Rcpp::wrap(std::get<0>(tup)));
584 }
585
586 //FGWR and FWR
587 else
588 {
589 return Rcpp::List::create(); // no partial residuals here
590 }
591 }, p_res);
592}

607template< typename INPUT, typename OUTPUT >
608 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>)
609Rcpp::List
610wrap_prediction_to_R_list(const std::vector< std::vector<OUTPUT>> &pred,
611 const std::vector< INPUT> &abscissa,
612 const FDAGWR_TRAITS::Dense_Matrix &pred_coeff,
613 std::string basis_t,
614 std::size_t n_basis,
615 std::size_t basis_deg,
616 const FDAGWR_TRAITS::Dense_Matrix &basis_knots)
617{
618 Rcpp::List pred_w;
619
620 //evaluation part
621 std::size_t number_pred = pred.size();
622 Rcpp::List predictions(number_pred);
623 Rcpp::CharacterVector names(number_pred);
624
625 for(std::size_t i = 0; i < number_pred; ++i){
626 predictions[i] = Rcpp::wrap(pred[i]);
627 names[i] = std::string("unit_pred_") + std::to_string(i+1);
628 }
629
630 predictions.names() = names;
631
632 //pred evaluation
633 Rcpp::List pred_ev;
634 pred_ev[std::string{FDAGWR_HELPERS_for_PRED_NAMES::pred} + "_ev"] = predictions;
635 pred_ev[std::string{FDAGWR_HELPERS_for_PRED_NAMES::abscissa} + "_ev"] = abscissa;
636 //pred fd
637 Rcpp::List pred_fd;
638 pred_fd[std::string{FDAGWR_HELPERS_for_PRED_NAMES::pred} + "_" + std::string{FDAGWR_HELPERS_for_PRED_NAMES::coeff_basis}] = Rcpp::wrap(pred_coeff);
639 pred_fd[std::string{FDAGWR_HELPERS_for_PRED_NAMES::pred} + "_" + std::string{FDAGWR_HELPERS_for_PRED_NAMES::basis_t}] = basis_t;
640 pred_fd[std::string{FDAGWR_HELPERS_for_PRED_NAMES::pred} + "_" + std::string{FDAGWR_HELPERS_for_PRED_NAMES::n_basis}] = n_basis;
641 pred_fd[std::string{FDAGWR_HELPERS_for_PRED_NAMES::pred} + "_" + std::string{FDAGWR_HELPERS_for_PRED_NAMES::basis_deg}] = basis_deg;
642 pred_fd[std::string{FDAGWR_HELPERS_for_PRED_NAMES::pred} + "_" + std::string{FDAGWR_HELPERS_for_PRED_NAMES::basis_knots}] = Rcpp::wrap(basis_knots);
643
644 //returning everything
645 pred_w["evaluation"] = pred_ev;
646 pred_w["fd"] = pred_fd;
647
648 return pred_w;
649}
