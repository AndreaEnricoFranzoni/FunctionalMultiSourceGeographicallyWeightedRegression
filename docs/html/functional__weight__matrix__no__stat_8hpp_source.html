154 inline
155 void
156 computing_weights()
157 {
158 //to shared the values with OMP
159 auto n_stat_units = this->n();
160 //preparing the container for the functional non-stationary weight matrix
161 m_weights.resize(n_stat_units);
162
163#ifdef _OPENMP
164#pragma omp parallel for shared(m_distance_matrix,n_stat_units) num_threads(this->number_threads())
165#endif
166 for(std::size_t i = 0; i < n_stat_units; ++i)
167 {
168 //non stationary weights with respect to unit i-th
169
170 //Eigen vector with the distances with respect to unit i-th
171 auto weights_non_stat_unit_i = m_distance_matrix[i];
172
173 //applying the kernel function to correctly smoothing the distances
174 std::transform(weights_non_stat_unit_i.data(),
175 weights_non_stat_unit_i.data() + weights_non_stat_unit_i.size(),
176 weights_non_stat_unit_i.data(),
177 [this](double dist){return this->kernel_eval(dist,this->m_kernel_bandwith);});
178
179 //preparing the container for the functional non-stationary matrix of unit i-th
180 std::vector< F_OBJ > weights_unit_i;
181 weights_unit_i.reserve(n_stat_units);
182
183 //computing the interaction within kernel application to distances and response reconstruction, unit i-th and all the other ones
184 for (std::size_t j = 0; j < n_stat_units; ++j)
185 {
186 double alpha_i_j = weights_non_stat_unit_i[j];
187 F_OBJ w_i_j = [j,alpha_i_j,this](F_OBJ_INPUT loc){ return alpha_i_j * this->y_recostruction_weights_fd().eval(loc,j);};
188 weights_unit_i.push_back(w_i_j);
189 }
190
191 //storing the functional non-stationary matrix for unit i-th (corresponding to index unit_index)
192 m_weights[i] = weights_unit_i;
193 }
194 }

199 inline
200 void
201 compute_weights_pred()
202 {
203 std::size_t n_train = m_distance_matrix_pred.n_train();
204 std::size_t n_pred = m_distance_matrix_pred.n_pred();
205
206 m_weights.resize(n_pred);
207
208#ifdef _OPENMP
209#pragma omp parallel for shared(m_distance_matrix_pred,n_train,n_pred) num_threads(this->number_threads())
210#endif
211 for(std::size_t i_pred = 0; i_pred < n_pred; ++i_pred)
212 {
213 std::vector<double> weights_non_stat_unit_i_pred = m_distance_matrix_pred.distances()[i_pred];
214
215 //applying the kernel function to correctly smoothing the distances
216 std::transform(weights_non_stat_unit_i_pred.begin(),
217 weights_non_stat_unit_i_pred.end(),
218 weights_non_stat_unit_i_pred.begin(),
219 [this](double dist){return this->kernel_eval(dist,this->m_kernel_bandwith);});
220
221 //preparing the container for the functional non-stationary matrix of pred unit i_pred-th
222 std::vector< F_OBJ > weights_unit_i_pred;
223 weights_unit_i_pred.reserve(n_train);
224
225 //computing the interaction within kernel application to distances and response reconstruction, unit i-th and all the other ones
226 for (std::size_t j_train = 0; j_train < n_train; ++j_train)
227 {
228 double alpha_i_j = weights_non_stat_unit_i_pred[j_train];
229 F_OBJ w_i_j = [j_train,alpha_i_j,this](F_OBJ_INPUT loc){ return alpha_i_j * this->y_recostruction_weights_fd().eval(loc,j_train);};
230 weights_unit_i_pred.push_back(w_i_j);
231 }
232
233 //storing the functional non-stationary matrix for unit i-th (corresponding to index unit_index)
234 m_weights[i_pred] = weights_unit_i_pred;
235 }
236 }
