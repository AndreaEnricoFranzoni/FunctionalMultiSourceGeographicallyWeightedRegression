131 inline
132 std::vector< std::vector<OUTPUT>>
133 evalPred(const functional_matrix<INPUT,OUTPUT> &pred,
134 const std::vector<INPUT> &abscissa)
135 const
136 {
137 assert(pred.cols() == 1);
138 std::size_t n_pred = pred.rows();
139 std::size_t n_abs = abscissa.size();
140
141 std::vector< std::vector<OUTPUT>> evaluations_pred;
142 evaluations_pred.resize(n_pred);
143
144#ifdef _OPENMP
145#pragma omp parallel for shared(pred,n_pred,abscissa,n_abs) num_threads(this->number_threads())
146#endif
147 for(std::size_t i = 0; i < n_pred; ++i)
148 {
149 std::vector<OUTPUT> evaluations_pred_i;
150 evaluations_pred_i.resize(n_abs);
151
152 //evaluating the i-th predicted unit
153 std::transform(abscissa.cbegin(),
154 abscissa.cend(),
155 evaluations_pred_i.begin(),
156 [&pred,i](fm_utils::input_param_t<FUNC_OBJ<INPUT,OUTPUT>> x){return pred(i,0)(x);});
157
158 evaluations_pred[i] = evaluations_pred_i;
159 }
160
161 return evaluations_pred;
162 }
