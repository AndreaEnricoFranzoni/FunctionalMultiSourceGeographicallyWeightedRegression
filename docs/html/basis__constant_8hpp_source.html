132 inline

134 smoothing(const FDAGWR_TRAITS::Dense_Matrix & f_ev,
135 const FDAGWR_TRAITS::Dense_Matrix & knots)
136 const
137 override
138 {
139 assert((f_ev.rows() == knots.rows()) && (f_ev.cols() == 1) && (knots.cols() == 1));
140
141 //for constant basis, the smoothing is simply the mean value of the function along the domain
142 std::size_t number_knots = knots.rows();
143 FDAGWR_TRAITS::Dense_Matrix c = FDAGWR_TRAITS::Dense_Matrix::Zero(1,1);
144 for(std::size_t i = 0; i < number_knots; ++i){ c(0,0) += f_ev(i,0);}
145
146 return c/number_knots;
147 }
