149 template <class T>
150 functional_matrix(const Expr<T,INPUT,OUTPUT> &e)
151 : m_data()
152 {
153 //casting
154 const T &et(e);
155 m_rows = et.rows();
156 m_cols = et.cols();
157 m_data.reserve(et.size());
158 //storing column-wise
159 for (std::size_t j = 0; j < et.cols(); ++j){
160 for(std::size_t i = 0; i < et.rows(); ++i){
161 m_data.emplace_back(et(i,j));}}
162 }

308 row_sub(const std::vector< F_OBJ > &new_row, std::size_t idx)
309 {
310 assert(new_row.size() == m_cols);

341 void
342 transposing()
343 {
344 if(m_rows!=static_cast<std::size_t>(1) && m_cols!=static_cast<std::size_t>(1))
345 {
346 std::vector< F_OBJ > temp;
347 temp.resize(this->size());
348
349#ifdef _OPENMP
350#pragma omp parallel for collapse(2) shared(m_rows,m_cols,temp,m_data) num_threads(8)
351#endif
352 for (std::size_t i = 0; i < m_rows; ++i)
353 {
354 for (std::size_t j = 0; j < m_cols; ++j)
355 {
356 temp[i*m_cols + j] = m_data[j*m_rows + i]; //swaps appropriately
357 }
358 }
359 //swap them
360 std::swap(m_data,temp);
361 temp.clear();
362 }
363 //swapping number of rows and cols
364 std::swap(m_rows,m_cols);
365 }

528template< typename INPUT = double, typename OUTPUT = double >
529 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>)
530inline
531functional_matrix<INPUT,OUTPUT>
532scalar_to_functional(const Eigen::Matrix< OUTPUT, Eigen::Dynamic, Eigen::Dynamic > &Ms)
533{
534 //type stored by the functional matrix
535 using F_OBJ = FUNC_OBJ<INPUT,OUTPUT>;
536 //input type of the elements of the functional matrix
537 using F_OBJ_INPUT = fm_utils::input_param_t<F_OBJ>;
538 //function to convert a scalar into a constant function
539 std::function< F_OBJ(const OUTPUT &) > scalar_to_const_func = [](const OUTPUT &a){ return [a](F_OBJ_INPUT x){return static_cast<OUTPUT>(a);};};
540
541 //reshaping the scalar matrix (.reshaped() aligns column-wise!) in order to iterate along it
542 auto R = Ms.reshaped();
543 //container to store the transformation from scalar to f
544 std::vector< F_OBJ > f_vec;
545 f_vec.resize(Ms.size());
546 //transforming each element of S into a constant function
547 std::transform(R.cbegin(),R.cend(),f_vec.begin(),scalar_to_const_func);
548 //constructing the functional matrix
549 functional_matrix<INPUT,OUTPUT> Mf(f_vec,Ms.rows(),Ms.cols());
550
551 return Mf;
552}
