39template< DISTANCE_MEASURE distance_measure >
40double
41distance_matrix_pred<distance_measure>::pointwise_distance(std::size_t loc_i_pred,
42 std::size_t loc_j_train,
43 DISTANCE_MEASURE_T<DISTANCE_MEASURE::EUCLIDEAN>)
44const
45{
46 // given coordinates of unit loc_i and loc_j, doing, for each coordinate, difference and squared.
47 // Square root of the sum of the previous quantities
48 return std::sqrt((m_coordinates_pred.row(loc_i_pred).array() - m_coordinates_train.row(loc_j_train).array()).square().sum());
49}
