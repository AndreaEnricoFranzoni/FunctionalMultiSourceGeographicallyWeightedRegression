164 template <class T>
165 functional_matrix_sparse(const Expr<T,INPUT,OUTPUT> &e)
166 : m_data()
167 {
168 //casting
169 const T &et(e);
170 m_rows = et.rows();
171 m_cols = et.cols();
172 m_nnz = et.size();
173
174 //reserving the correct amount of capacity
175 m_data.reserve(et.size());
176 m_rows_idx.reserve(et.size());
177 m_cols_idx.reserve(et.cols()+1);
178 m_cols_idx.emplace_back(static_cast<std::size_t>(0)); //the first element of the cumulative number of elements per columns is always 0
179 //counter for the cumulative number of elements along columns
180 std::size_t counter_cols_elem = 0;
181
182 //looping as this for column-wise storage
183 for(std::size_t j = 0; j < et.cols(); ++j){
184 for(std::size_t i = 0; i < et.rows(); ++i){
185 //inserting only if non-null: IT LACKS THIS IF CONDITION (NOW IT COMPARE THE ADDRESS OF THE NON CONST NULL FUNCTION)
186 m_data.emplace_back(et(i,j));
187 if(&m_data.back() == &functional_matrix_sparse<INPUT,OUTPUT>::m_null_function_non_const)
188 {
189 m_data.pop_back();
190 }
191 else
192 {
193 m_rows_idx.emplace_back(i);
194 counter_cols_elem += 1;
195 }
196 }
197 m_cols_idx.emplace_back(counter_cols_elem);
198 }
199 }

254 bool
255 check_row_presence(std::size_t idx)
256 const
257 {
258 //checking that the passed index is coherent with the matrix dimension
259 assert(idx < m_rows);
260 //it is sufficient that in the vector containing the rows idx there is once idx: need to go with std::find instead of std::binary_search since elements are not ordered
261 return std::find(m_rows_idx.cbegin(),m_rows_idx.cend(),idx) != m_rows_idx.cend();
262 }

269 bool
270 check_col_presence(std::size_t idx)
271 const
272 {
273 //checking that the passed index is coherent with the matrix dimension
274 assert(idx < m_cols);
275 //it is sufficient that between col i-th and i+1-th there is an increment in number of elements
276 return m_cols_idx[idx] < m_cols_idx[idx+1];

285 bool
286 check_elem_presence(std::size_t i, std::size_t j)
287 const
288 {
289 //checking if there row i and col j are present
290 if(!this->check_col_presence(j) || !this->check_row_presence(i)){ return false;}
291 //searching if within the row indeces of the col there is the one requested. Since, for each column, elements of m_rows_idx are ordered, it is possible to relay on std::binary_search
292 return std::binary_search(std::next(m_rows_idx.cbegin(),m_cols_idx[j]), //start of rows idx in col j
293 std::next(m_rows_idx.cbegin(),m_cols_idx[j+1]), //end of rows idx in col j
294 i); //row index that has to be in the row
295 }

304 F_OBJ &
305 operator()
306 (std::size_t i, std::size_t j)
307 {
308 //checking input consistency
309 assert(i < m_rows && j < m_cols);
310 //check the col presence
311 if(!this->check_col_presence(j)) { return this->m_null_function_non_const;}
312 //looking at the position of row of the element in the range indicated by the right column. Since, for each column, elements of m_rows_idx are ordered, it is possible to relay on std::lower_bound (finds the first element, in an ordered range, >= value)
313 auto elem = std::lower_bound(std::next(m_rows_idx.cbegin(),m_cols_idx[j]),
314 std::next(m_rows_idx.cbegin(),m_cols_idx[j+1]),
315 i);
316 //taking the distance from the begin to retrain the position in the value's vector, if present, null function if not (since finds the first element, in an ordered range, >= value, it is necessary to check it also)
317 return (elem!=std::next(m_rows_idx.cbegin(),m_cols_idx[j+1]) && *elem==i) ? m_data[std::distance(m_rows_idx.cbegin(),elem)] : this->m_null_function_non_const;
318 }

411 void
412 transposing()
413 {
414 //row vector ==> col vector
415 if(m_rows == static_cast<std::size_t>(1) && m_cols != static_cast<std::size_t>(1))
416 {
417 //m_data does not change
418
419 //m_rows_idx: looking in m_cols_idx where there is an increase
420 m_rows_idx.clear();
421 m_rows_idx.reserve(m_nnz);
422 for(std::size_t i = 1; i <= m_cols; ++i){ //looping from 1 since m_cols_idx has m_cols+1 elements, the first one being always 0
423 if(m_cols_idx[i] > m_cols_idx[i-1]){
424 m_rows_idx.emplace_back(i-1);}}
425 //m_cols_idx: only one column containing all the elements
426 m_cols_idx.clear();
427 m_cols_idx.reserve(2);
428 m_cols_idx.emplace_back(static_cast<std::size_t>(0));
429 m_cols_idx.emplace_back(m_nnz);
430 }
431
432 //col vector ==> row vector
433 if(m_cols == static_cast<std::size_t>(1) && m_rows != static_cast<std::size_t>(1))
434 {
435 //m_data does not change
436
437 //m_cols_idx: the old m_rows indeces indicates in which position there will be an increase by one in the new m_cols_idx
438 m_cols_idx.clear();
439 m_cols_idx.reserve(m_rows + 1);
440 m_cols_idx.emplace_back(static_cast<std::size_t>(0)); //first element is always 0
441 //loop solo sulle righe che ho, conto ogni quanto appare un elemento
442 std::vector<std::size_t> rows_difference = m_rows_idx;
443 rows_difference.push_back(m_rows);
444 std::adjacent_difference(rows_difference.begin(),rows_difference.end(),rows_difference.begin()); //il primo elemento del risultato di std::adjacent_difference sempre il primo elemento del range in input
445 //the first element is always 0
446 std::size_t el = m_cols_idx.back();
447 for(auto it : rows_difference){
448 //*it says how many elements have to be insert before increasing it
449 for (std::size_t ii = 0; ii < it; ++ii){
450 m_cols_idx.emplace_back(el);}
451 el += 1;}
452
453 //m_rows_idx are all 0s, since all the elements will be in the first (and only) row
454 m_rows_idx.clear();
455 m_rows_idx.resize(m_nnz);
456 std::fill(m_rows_idx.begin(),m_rows_idx.end(),static_cast<std::size_t>(0));
457 }
458
459 //general matrix ==> its transpost
460 if(m_cols != static_cast<std::size_t>(1) && m_rows != static_cast<std::size_t>(1))
461 {
462 //new container for data
463 std::vector< F_OBJ > temp_data;
464 temp_data.reserve(this->size());
465 //new container for row_idx
466 std::vector<std::size_t> temp_row_idx;
467 temp_row_idx.reserve(this->size());
468 //new container for col_idx
469 std::vector<std::size_t> temp_col_idx;
470 temp_col_idx.reserve(this->rows() + 1);
471 temp_col_idx.emplace_back(static_cast<std::size_t>(0));
472
473 //loop su tutte le righe dell'ogetto originale, che diventeranno le colonne del trasposto
474 for(std::size_t i = 0; i < this->rows(); ++i)
475 {
476 //conto quanti elementi ci sono nella vecchia riga i-th, quindi nuova colonna
477 std::size_t counter_el_row_i = 0;
478 //only if the row i-th is present
479 if(this->check_row_presence(i))
480 {
481 //trovo tutti gli elementi che sono nella riga i-th
482 for (std::size_t ii = 0; ii < m_nnz; ++ii)
483 {
484 //se l'elemento nella riga i-th
485 if(m_rows_idx[ii] == i)
486 {
487 //salvo(riga->col: sto salvando per colonne)
488 temp_data.emplace_back(m_data[ii]);
489 //un elemento in pi in quella colonna
490 counter_el_row_i += static_cast<std::size_t>(1);
491 //come i nuovi indici di riga: la posizione ii mi indica dove salvo effettivamente l'elemento nel vettore
492 //dunque l'elemento ii+1. Devo trovare il primo elemento negli indici di colonna che >=(ii+1), e la sua posizione in m_cols_idx,
493 //tolto 1, mi d la colonna in cui salvato, dunque la riga nella trasposta
494 temp_row_idx.emplace_back(std::distance(m_cols_idx.cbegin(),std::lower_bound(m_cols_idx.cbegin(),m_cols_idx.cend(),ii+1) - 1));
495 }
496 }
497 }
498 //salvo quanti elementi in ogni colonna
499 temp_col_idx.emplace_back(temp_col_idx.back() + counter_el_row_i);
500 }
501 //swap operations
502 std::swap(m_data,temp_data);
503 std::swap(temp_row_idx,m_rows_idx);
504 std::swap(temp_col_idx,m_cols_idx);
505 }
506 //swap number of rows and cols
507 std::swap(m_cols,m_rows);
508 }
