51template< typename INPUT = double, typename OUTPUT = double, class domain_type = FDAGWR_TRAITS::basis_geometry, template <typename> class basis_type = bsplines_basis >
52 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>) && fdagwr_concepts::as_interval<domain_type> && fdagwr_concepts::as_basis<basis_type<domain_type>>
53inline
54functional_matrix<INPUT,OUTPUT>
55wrap_into_fm(const functional_data<domain_type,basis_type> &fd,
56 int number_threads,
57 bool as_column = true)
58{
59 //std::function object stored
60 using F_OBJ = FUNC_OBJ<INPUT,OUTPUT>;
61 //std::function input type
62 using F_OBJ_INPUT = fm_utils::input_param_t<F_OBJ>;
63
64 std::vector< F_OBJ > f;
65 f.resize(fd.n());
66
67#ifdef _OPENMP
68#pragma omp parallel for shared(fd) num_threads(number_threads)
69#endif
70 for(std::size_t unit_i = 0; unit_i < fd.n(); ++unit_i)
71 {
72 f[unit_i] = [unit_i,&fd](F_OBJ_INPUT x){return fd.eval(x,unit_i);};
73 }
74
75 if(as_column == true)
76 {
77 functional_matrix<INPUT,OUTPUT> fm(std::move(f),fd.n(),1);
78 return fm;
79 }
80 else
81 {
82 functional_matrix<INPUT,OUTPUT> fm(std::move(f),1,fd.n());
83 return fm;
84 }
85}

100template< typename INPUT = double, typename OUTPUT = double, class domain_type = FDAGWR_TRAITS::basis_geometry, FDAGWR_COVARIATES_TYPES stationarity_t = FDAGWR_COVARIATES_TYPES::STATIONARY >
101 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>) && fdagwr_concepts::as_interval<domain_type>
102inline
103functional_matrix<INPUT,OUTPUT>
104wrap_into_fm(const functional_data_covariates<domain_type,stationarity_t> &X,
105 int number_threads)
106{
107 //std::function object stored
108 using F_OBJ = FUNC_OBJ<INPUT,OUTPUT>;
109 //std::function input type
110 using F_OBJ_INPUT = fm_utils::input_param_t<F_OBJ>;
111
112 std::vector< F_OBJ > f;
113 f.resize(X.n()*X.q());
114
115#ifdef _OPENMP
116#pragma omp parallel for collapse(2) shared(X) num_threads(number_threads)
117#endif
118 for (std::size_t unit_i = 0; unit_i < X.n(); ++unit_i)
119 {
120 for (std::size_t cov_j = 0; cov_j < X.q(); ++cov_j)
121 {
122 f[cov_j*X.n() + unit_i] = [unit_i,cov_j,&X](F_OBJ_INPUT x){return X.eval(x,cov_j,unit_i);};
123 }
124 }
125
126 functional_matrix<INPUT,OUTPUT> fm(std::move(f),X.n(),X.q());
127 return fm;
128}

146template< typename INPUT = double, typename OUTPUT = double, class domain_type = FDAGWR_TRAITS::basis_geometry, template <typename> class basis_type = bsplines_basis >
147 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>) && fdagwr_concepts::as_interval<domain_type> && fdagwr_concepts::as_basis<basis_type<domain_type>>
148inline
149functional_matrix_sparse<INPUT,OUTPUT>
150wrap_into_fm(const basis_type<domain_type> &bs,
151 std::size_t n,
152 std::size_t L)
153{
154 //std::function object stored
155 using F_OBJ = FUNC_OBJ<INPUT,OUTPUT>;
156 //std::function input type
157 using F_OBJ_INPUT = fm_utils::input_param_t<F_OBJ>;
158
159 //one row for each unit
160 std::size_t rows = n;
161 //the total number of cols is the product within the number of units and the number of basis
162 std::size_t cols = n*L;
163 //the construction of the matrix by block, only one element for each column
164 std::size_t nnz = n*L;
165
166 //containers storing elements and indices
167 std::vector< F_OBJ > f;
168 f.reserve(nnz);
169 std::vector<std::size_t> row_idx;
170 row_idx.reserve(nnz);
171 std::vector<std::size_t> col_idx;
172 col_idx.resize(cols + 1);
173 std::iota(col_idx.begin(),col_idx.end(),static_cast<std::size_t>(0)); //cumulative number of elements in the cols is simply an increasing count of naturals
174
175 //filling f
176 for (std::size_t unit_i = 0; unit_i < n; ++unit_i){
177 for (std::size_t l_i = 0; l_i < L; ++l_i){
178 row_idx.emplace_back(unit_i);
179 f.emplace_back([l_i,&bs](F_OBJ_INPUT x){return bs.eval_base(x)(0,l_i);});}}
180
181 functional_matrix_sparse<INPUT,OUTPUT> fm(f,rows,cols,row_idx,col_idx);
182 return fm;
183}

197template< typename INPUT = double, typename OUTPUT = double, class domain_type = FDAGWR_TRAITS::basis_geometry, template <typename> class basis_type = bsplines_basis >
198 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>) && fdagwr_concepts::as_interval<domain_type> && fdagwr_concepts::as_basis<basis_type<domain_type>>
199inline
200functional_matrix_sparse<INPUT,OUTPUT>
201wrap_into_fm(const basis_systems<domain_type,basis_type> &bs)
202{
203 //std::function object stored
204 using F_OBJ = FUNC_OBJ<INPUT,OUTPUT>;
205 //std::function input type
206 using F_OBJ_INPUT = fm_utils::input_param_t<F_OBJ>;
207
208 //one row for each covariate
209 std::size_t rows = bs.q();
210 //the total number of cols is the sum of the number of basis of all the covariates
211 std::size_t cols = std::reduce(bs.numbers_of_basis().cbegin(),bs.numbers_of_basis().cend(),static_cast<std::size_t>(0));
212 //the construction of the matrix by block, only one element for each column
213 std::size_t nnz = cols;
214
215 //containers storing elements and indices
216 std::vector< F_OBJ > f;
217 f.reserve(nnz);
218 std::vector<std::size_t> row_idx;
219 row_idx.reserve(nnz);
220 std::vector<std::size_t> col_idx;
221 col_idx.resize(cols + 1);
222 std::iota(col_idx.begin(),col_idx.end(),static_cast<std::size_t>(0)); //cumulative number of elements in the cols is simply an increasing count of naturals
223
224 //filling f
225 for (std::size_t cov_i = 0; cov_i < bs.q(); ++cov_i){
226 for(std::size_t base_j = 0; base_j < bs.numbers_of_basis()[cov_i]; ++base_j){
227 //row cov_i-th contains a number of elements equal to the number of basis for the covariate cov_i-th
228 row_idx.push_back(cov_i);
229 //storing the basis accordingly to the type
230 f.emplace_back([cov_i,base_j,&bs](F_OBJ_INPUT x){return bs.systems_of_basis()[cov_i].eval_base(x)(0,base_j);});}}
231
232 functional_matrix_sparse<INPUT,OUTPUT> fm(f,rows,cols,row_idx,col_idx);
233 return fm;
234}

247template< typename INPUT = double, typename OUTPUT = double, class domain_type = FDAGWR_TRAITS::basis_geometry, template <typename> class basis_type = bsplines_basis, FDAGWR_COVARIATES_TYPES stationarity_t = FDAGWR_COVARIATES_TYPES::STATIONARY >
248 requires (std::integral<INPUT> || std::floating_point<INPUT>) && (std::integral<OUTPUT> || std::floating_point<OUTPUT>) && fdagwr_concepts::as_interval<domain_type> && fdagwr_concepts::as_basis<basis_type<domain_type>>
249inline
250functional_matrix_diagonal<INPUT,OUTPUT>
251wrap_into_fm(const functional_weight_matrix_stationary<INPUT,OUTPUT,domain_type,basis_type,stationarity_t> &W,
252 int number_threads)
253{
254 static_assert(stationarity_t == FDAGWR_COVARIATES_TYPES::STATIONARY,
255 "Functional weight matrix for stationary covariates needs FDAGWR_COVARIATES_TYPES::STATIONARY as template parameter");
256
257 //std::function object stored
258 using F_OBJ = FUNC_OBJ<INPUT,OUTPUT>;
259
260 std::vector< F_OBJ > f;
261 std::size_t n = W.n();
262 f.resize(n);
263
264#ifdef _OPENMP
265#pragma omp parallel for shared(n,W) num_threads(number_threads)
266#endif
267 for(std::size_t unit_i = 0; unit_i < n; ++unit_i)
268 {
269 f[unit_i] = W.weights()[unit_i];
270 }
271
272 functional_matrix_diagonal<INPUT,OUTPUT> fm(f,n);
273 return fm;
274}

287template< typename INPUT = double, typename OUTPUT = double, class domain_type = FDAGWR_TRAITS::basis_geometry, template <typename> class basis_type = bsplines_basis, FDAGWR_COVARIATES_TYPES stationarity_t = FDAGWR_COVARIATES_TYPES::NON_STATIONARY >

289inline
290std::vector< functional_matrix_diagonal<INPUT,OUTPUT> >
291wrap_into_fm(const functional_weight_matrix_non_stationary<INPUT,OUTPUT,domain_type,basis_type,stationarity_t> &W,
292 int number_threads)
293{
294 static_assert(stationarity_t == FDAGWR_COVARIATES_TYPES::NON_STATIONARY ||
295 stationarity_t == FDAGWR_COVARIATES_TYPES::EVENT ||
296 stationarity_t == FDAGWR_COVARIATES_TYPES::STATION,
297 "Functional weight matrix for non stationary covariates needs FDAGWR_COVARIATES_TYPES::NON_STATIONARY or FDAGWR_COVARIATES_TYPES::EVENT or FDAGWR_COVARIATES_TYPES::STATION as template parameter");
298
299 //std::function object stored
300 using F_OBJ = FUNC_OBJ<INPUT,OUTPUT>;
301
302 std::size_t n_matrices = W.weights().size();
303 std::size_t n = W.n(); //number of statistical units
304 //storing all the functional diagonal matrices
305 std::vector<functional_matrix_diagonal<INPUT,OUTPUT>> fm;
306 fm.reserve(n_matrices);
307
308 for(std::size_t i = 0; i < n_matrices; ++i)
309 {
310 std::vector< F_OBJ > f;
311 f.resize(n);
312
313#ifdef _OPENMP
314#pragma omp parallel for shared(i,n,W) num_threads(number_threads)
315#endif
316 for(std::size_t unit_i = 0; unit_i < n; ++unit_i)
317 {
318 f[unit_i] = W.weights()[i][unit_i];
319 }
320
321 //fm.emplace_back(std::move(f),n);
322 //fm.emplace_back(f,n);
323 functional_matrix_diagonal<INPUT,OUTPUT> fm_i(f,n);
324 fm.push_back(fm_i);
325 }
326
327 return fm;
328}
