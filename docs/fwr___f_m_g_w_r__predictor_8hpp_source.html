279 computePartialResiduals()
280 override
281 {
282 if (!this->in_cascade_estimation())
283 {
284 //y_tilde_new(t) n_trainx1
285 m_y_tilde_new = fm_prod(m_phi,m_c_tilde_hat);

299 inline
300 void
301 computeBNew(const std::map<std::string,std::vector< functional_matrix_diagonal<INPUT,OUTPUT> >> &W)
302 override
303 {
304 assert(W.size() == 1);
305 auto Wnc_new = W.at(std::string{fwr_FMGWR_predictor<INPUT,OUTPUT>::id_NC});
306 //input coherency
307 for(std::size_t i = 0; i < Wnc_new.size(); ++i){
308 assert((Wnc_new[i].rows() == this->n_train()) && (Wnc_new[i].cols() == this->n_train()));}
309 //number of units to be predicted
310 std::size_t n_pred = Wnc_new.size();
311
312
313 //compute the non-stationary betas in the new locations
314 //penalties in the new locations
315 //(j_tilde + Rnc)^-1 (LncxLnc)
316 std::vector< Eigen::PartialPivLU<FDAGWR_TRAITS::Dense_Matrix> > j_tilde_Rnc_inv = this->operator_comp().compute_penalty(m_eta_t,m_Xnc_train_t,Wnc_new,m_Xnc_train,m_eta,m_Rnc); //per applicarlo: j_double_tilde_RE_inv[i].solve(M) equivale a ([J_i_tilde_tilde + Re]^-1)*M
317 //COMPUTING all the m_bnc in the new locations, SO THE COEFFICIENTS FOR THE BASIS EXPANSION OF THE NON-STATIONARY BETAS (n_pred elements Lncx1)
318 m_bnc_pred = this->operator_comp().compute_operator(m_eta_t,m_Xnc_train_t,Wnc_new,m_y_tilde_new,j_tilde_Rnc_inv);
319
320
321 //
322 //wrapping the b from the shape useful for the computation into a more useful format for reporting the results: TENERE
323 //
324 //non-stationary covariates (qnc elements of n_pred elements 1xLnc_i)
325 m_Bnc_pred = this->operator_comp().wrap_operator(m_bnc_pred,m_Lnc_j,m_qnc,n_pred);
326
327 }

366 inline
367 functional_matrix<INPUT,OUTPUT>
368 predict(const std::map<std::string,functional_matrix<INPUT,OUTPUT>> &X_new)
369 const
370 override
371 {
372 //input coherence
373 assert(X_new.size() == 2);
374
375 auto Xc_new = X_new.at(std::string{fwr_FMGWR_predictor<INPUT,OUTPUT>::id_C});
376 auto Xnc_new = X_new.at(std::string{fwr_FMGWR_predictor<INPUT,OUTPUT>::id_NC});
377
378 assert(Xc_new.rows() == Xnc_new.rows());
379 std::size_t n_pred = Xc_new.rows();
380 assert(n_pred == m_BetaNC.size());
381 assert((Xc_new.cols() == m_qc) && (Xnc_new.cols() == m_qnc));
382
383
384 //y_new = X_new*beta = Xc_new*beta_c + Xnc_new*beta_nc
385 functional_matrix<INPUT,OUTPUT> y_new_C = fm_prod(Xc_new,m_BetaC,this->number_threads()); //n_pred x 1
386 functional_matrix<INPUT,OUTPUT> y_new_NC(n_pred,1);
387
388
389#ifdef _OPENMP
390#pragma omp parallel for shared(Xnc_new,m_BetaNC,y_new_NC,n_pred,m_qnc) num_threads(this->number_threads())
391#endif
392 for(std::size_t i = 0; i < n_pred; ++i)
393 {
394 std::vector< FUNC_OBJ<INPUT,OUTPUT> > xnc_new_i(Xnc_new.row(i).cbegin(),Xnc_new.row(i).cend()); //1xqnc
395 functional_matrix<INPUT,OUTPUT> Xnc_new_i(xnc_new_i,1,m_qnc);
396 y_new_NC(i,0) = fm_prod(Xnc_new_i,m_BetaNC[i],this->number_threads())(0,0);
397 }
398
399 return y_new_C + y_new_NC;
400 }
