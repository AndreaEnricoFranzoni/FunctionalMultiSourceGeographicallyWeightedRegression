65 inline
66 FDAGWR_TRAITS::Dense_Matrix
67 fm_integration(const functional_matrix<INPUT,OUTPUT> &integrand)
68 const
69 {
70 //storing integrals results
71 std::vector<OUTPUT> result_integrand;
72 result_integrand.resize(integrand.size());
73
74#ifdef _OPENMP
75#pragma omp parallel for shared(integrand,m_integrating,result_integrand) num_threads(m_number_threads)
76#endif
77 for(std::size_t i = 0; i < integrand.size(); ++i){
78 result_integrand[i] = m_integrating.integrate(integrand.as_vector()[i]);}
79
80 FDAGWR_TRAITS::Dense_Matrix result_integration = Eigen::Map< FDAGWR_TRAITS::Dense_Matrix >(result_integrand.data(), integrand.rows(), integrand.cols());
81
82 return result_integration;
83 }
