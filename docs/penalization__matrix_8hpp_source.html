84 template< typename BASIS_SPACE >
85 penalization_matrix(BASIS_SPACE&& bs,
86 const std::vector<double>& lambdas)
87 :
88 m_q(bs.q()),
89 m_Lj(bs.numbers_of_basis()),
90 m_L(std::reduce(bs.numbers_of_basis().cbegin(),bs.numbers_of_basis().cend(),static_cast<std::size_t>(0)))
91 {
92 //storing the penalty for each basis system in an Eigen::Triplet
93 std::vector<Eigen::Triplet<double>> penalty_matrices_triplets;
94 //the unlikely scenario in which all the L2 scalar products are not-null
95 penalty_matrices_triplets.reserve(std::transform_reduce(m_Lj.cbegin(),
96 m_Lj.cend(),
97 static_cast<std::size_t>(0),
98 std::plus{},
99 [](const double &nb){return std::pow(nb,2);}));
100
101 //constructing the penalty matrices
102 for(std::size_t i = 0; i < m_q; ++i){
103
104 //penalties, for each basis system: PenaltyPolicy indicates the order
105 penalty_computation<PenaltyPolicy> penalty_comp;
106 FDAGWR_TRAITS::Sparse_Matrix PenaltyBasis_i = penalty_comp(bs,i);
107 PenaltyBasis_i *= lambdas[i];
108
109 //all the penalty matrix are squared matrices: therse are the index at which each block starts
110 std::size_t start_of_block = std::reduce(bs.numbers_of_basis().cbegin(),bs.numbers_of_basis().cbegin()+i,static_cast<std::size_t>(0));
111 //storing the matrix in the a vector of Eigen::Triplets
112 for (std::size_t k = 0; k < PenaltyBasis_i.outerSize(); ++k){
113 for (FDAGWR_TRAITS::Sparse_Matrix::InnerIterator it(PenaltyBasis_i,k); it; ++it){
114 penalty_matrices_triplets.emplace_back(it.row() + start_of_block,
115 it.col() + start_of_block,
116 it.value());}}}
117
118 //shrinking
119 penalty_matrices_triplets.shrink_to_fit();
120
121 //size for the penalization matrix
122 m_PenalizationMatrix.resize(m_L,m_L);
123
124 //constructing the penalization matrix as a sparse block matrix
125 m_PenalizationMatrix.setFromTriplets(penalty_matrices_triplets.begin(),penalty_matrices_triplets.end());
126 }
127
132 const FDAGWR_TRAITS::Sparse_Matrix& PenalizationMatrix() const {return m_PenalizationMatrix;}
133
138 std::vector<std::size_t> Lj() const {return m_Lj;}
139
144 std::size_t L() const {return m_L;}
145};
