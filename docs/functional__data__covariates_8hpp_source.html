44template< typename domain_type = FDAGWR_TRAITS::basis_geometry, FDAGWR_COVARIATES_TYPES stationarity_t = FDAGWR_COVARIATES_TYPES::STATIONARY >
45 requires fdagwr_concepts::as_interval<domain_type>
46class functional_data_covariates

49private:
51 std::size_t m_q;

68 functional_data_covariates(const std::vector<FDAGWR_TRAITS::Dense_Matrix> & coeff,
69 std::size_t q,
70 const std::vector<std::string> & basis_types,
71 const std::vector<std::size_t> & basis_degrees,
72 const std::vector<std::size_t> & basis_numbers,
73 const FDAGWR_TRAITS::Dense_Vector & knots,
74 const basis_factory::basisFactory& factoryBasis)
75 :
76 m_q(q)
77 {
78 //input coherency
79 m_n = m_q > 0 ? coeff[0].cols() : 0;
80
81 //creating the vector of fd
82 m_X.reserve(m_q);
83 for(std::size_t i = 0; i < m_q; ++i)
84 {
85 std::unique_ptr<basis_base_class<domain_type>> basis_i = factoryBasis.create(basis_types[i],knots,basis_degrees[i],basis_numbers[i]);
86 functional_data< domain_type,basis_base_class > x_i(std::move(coeff[i]),std::move(basis_i));
87 m_X.push_back(x_i);
88 }
89 }
90
94 std::size_t q() const {return m_q;}

99 std::size_t n() const {return m_n;}

108 double
109 eval(const double &loc, std::size_t cov_i, std::size_t unit_j)
110 const
111 {
112 return m_X[cov_i].eval(loc,unit_j); //evaluation of unit_j-th of covariate cov_i-th in location loc (starting from 0)
113 }
